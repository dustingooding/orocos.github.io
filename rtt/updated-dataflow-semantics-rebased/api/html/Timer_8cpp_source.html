61  void Timer::loop()
62  {
63  // This code is executed from mThread's thread
64  while (!mdo_quit) {
65  Time wake_up_time;
66  TimerId next_timer_id = 0;
67 
68  // Select next timer.
69  {// This scope is for MutexLock.
70  // find wake_up_time
71  // check timers queue.
72  MutexLock locker(m);
73  // We can't use infinite as the OS may internally use time_spec, which can not
74  // represent as much in the future (until 2038) // XXX Year-2038 Bug
75  wake_up_time = 1000000000LL * std::numeric_limits<int32_t>::max();
76  for (TimerIds::iterator it = mtimers.begin(); it != mtimers.end(); ++it) {
77  if ( it->expires != 0 && it->expires < wake_up_time ) {
78  wake_up_time = it->expires;
79  next_timer_id = it - mtimers.begin();
80  }
81  }
82  }// MutexLock
83 
84  // Wait
85  int ret = 0;
86  Time now = rtos_get_time_ns();
87  if ( wake_up_time > now )
88  ret = msem.waitUntil( wake_up_time ); // case of no timers or running timers
89  else
90  ret = -1; // case of timer overrun.
91 
92  // Timeout handling
93  if (ret == -1) {
94  // a timer expired
95  // expires: reset/reprogram the timer that expired:
96  {
97  MutexLock locker(m);
98  // detect corner case for resize:
99  if ( next_timer_id < int(mtimers.size()) ) {
100  // now clear or reprogram it.
101  TimerIds::iterator tim = mtimers.begin() + next_timer_id;
102  if ( tim->period ) {
103  // periodic timer
104  // if late by more than 4 periods, skip late updates
105  int maxDelayInPeriods = 4;
106  if (now - tim->expires > tim->period*maxDelayInPeriods) {
107  tim->expires += tim->period*((now - tim->expires) / tim->period);
108  }
109  tim->expires += tim->period;
110  } else {
111  // aperiodic timer
112  tim->expires = 0;
113  }
114  }
115  }
116 
117  // Second: notify waiting threads
118  {// This scope is for MutexLock.
119  MutexLock locker(m);
120  mtimers[next_timer_id].expired.broadcast();
121  }// MutexLock
122 
123  // Third: send the timeout signal and allow (within the callback)
124  // to reprogram the timer.
125  // If we would expires call timeout(), the code above would overwrite
126  // user settings.
127  timeout( next_timer_id );
128  }
129  }
130  }
