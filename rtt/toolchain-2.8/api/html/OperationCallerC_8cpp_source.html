62  void checkAndCreate() {
63  Logger::In in("OperationCallerC");
64  if ( ofp ) {
65  size_t sz = ofp->arity();
66  if ( sz == args.size() ) {
67  // may throw or return nill
68  m = ofp->produce(args, caller );
69  try {
70  s = ofp->produceSend(args, caller );
71  } catch( no_asynchronous_operation_exception const& /*e*/) {
72  // leave s empty.
73  }
74  args.clear();
75  if ( !m )
76  return;
77  if (rta)
78  try {
79  m = new DataSourceCommand( rta->updateAction( m.get() ) );
80  } catch( bad_assignment& /*ba*/ ) {
81  log(Error) << "Error in OperationCallerC::ret : can not convert return value of type "<< m->getType() << " to given type "<< rta->getType()<<endlog();
82  }
83 
84  }
85  }
86  }
