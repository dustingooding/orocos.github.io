74  void *thread_function(void* t)
75  {
79  Thread* task = static_cast<os::Thread*> (t);
80  Logger::In in(task->getName());
81 
82  task->configure();
83 
84  // signal to setup() that we're created.
85  rtos_sem_signal(&(task->sem));
86 
87  // This lock forces setup(), which holds the lock, to continue.
88  { MutexLock lock(task->breaker); }
89 #ifdef OROPKG_OS_THREAD_SCOPE
90  // order thread scope toggle bit on thread number
91  unsigned int bit = task->threadNumber();
92 #endif
93  SCOPE_OFF
94 
95  int overruns = 0, cur_sched = task->msched_type;
96  NANO_TIME cur_period = task->period;
97 
98  while (!task->prepareForExit)
99  {
100  TRY(
104  while (1)
105  {
106  if (!task->active || (task->active && task->period == 0) || !task->running )
107  {
108  // consider this the 'configuration or waiting for command state'
109  if (task->period != 0) {
110  overruns = 0;
111  // drop out of periodic mode:
112  rtos_task_set_period(task->getTask(), 0);
113  }
114  rtos_sem_wait(&(task->sem)); // wait for command.
115  task->configure(); // check for reconfigure
116  if (task->prepareForExit) // check for exit
117  {
118  break; // break while(1) {}
119  }
120  // end of configuration
121  }
122  // This is the running state. Only enter if a task is running
123  if ( task->running )
124  {
125  if (task->period != 0) // periodic
126  {
127  MutexLock lock(task->breaker);
128  while(task->running && !task->prepareForExit )
129  {
130  TRY
131  (
132  SCOPE_ON
133  task->step(); // one cycle
134  SCOPE_OFF
135  )
136  CATCH_ALL
137  (
138  SCOPE_OFF
139  throw;
140  )
141 
142  // Check changes in period
143  if ( cur_period != task->period) {
144  // reconfigure period before going to sleep
145  rtos_task_set_period(task->getTask(), task->period);
146  cur_period = task->period;
147  if (cur_period == 0)
148  break; // break while(task->running) if no longer periodic
149  }
150 
151  // Check changes in scheduler
152  if ( cur_sched != task->msched_type) {
153  rtos_task_set_scheduler(task->getTask(), task->msched_type);
154  cur_sched = task->msched_type;
155  }
156  // rtos_task_wait_period will return immediately if
157  // the task is not periodic (ie period == 0)
158  // return non-zero to indicate overrun.
159  if (rtos_task_wait_period(task->getTask()) != 0)
160  {
161  ++overruns;
162  if (overruns == task->maxOverRun)
163  break; // break while(task->running)
164  }
165  else if (overruns != 0)
166  --overruns;
167  } // while(task->running)
168  if (overruns == task->maxOverRun || task->prepareForExit)
169  break; // break while(1) {}
170  }
171  else // non periodic:
172  TRY
173  (
174  // this mutex guarantees that stop() waits
175  // until loop() returns.
176  MutexLock lock(task->breaker);
177 
178  task->inloop = true;
179  SCOPE_ON
180  task->loop();
181  SCOPE_OFF
182  task->inloop = false;
183  ) CATCH_ALL
184  (
185  SCOPE_OFF
186  task->inloop = false;
187  throw;
188  )
189  }
190  } // while(1)
191  if (overruns == task->maxOverRun)
192  {
193  task->emergencyStop();
194  Logger::In in(rtos_task_get_name(task->getTask()));
195  log(Critical) << rtos_task_get_name(task->getTask())
196  << " got too many periodic overruns in step() ("
197  << overruns << " times), stopped Thread !"
198  << endlog();
199  log() << " See Thread::setMaxOverrun() for info."
200  << endlog();
201  }
202  )CATCH(std::exception const& e,
203  SCOPE_OFF
204  task->emergencyStop();
205  Logger::In in(rtos_task_get_name(task->getTask()));
206  log(Critical) << rtos_task_get_name(task->getTask())
207  << " caught a C++ exception, stopped thread !"
208  << endlog();
209  log(Critical) << "exception was: "
210  << e.what() << endlog();
211  ) CATCH_ALL
212  (
213  SCOPE_OFF
214  task->emergencyStop();
215  Logger::In in(rtos_task_get_name(task->getTask()));
216  log(Critical) << rtos_task_get_name(task->getTask())
217  << " caught an unknown C++ exception, stopped thread !"
218  << endlog();
219  )
220  } // while (!prepareForExit)
221 
222  return 0;
223  }

337  bool Thread::start()
338  {
339  if ( period == 0)
340  {
341  // just signal if already active.
342  if ( isActive() ) {
343 #ifndef OROPKG_OS_MACOSX
344  // This is a 'weak' race condition.
345  // it could be that sem becomes zero
346  // after this check. Technically, this means
347  // loop is being executed (preemption) during start().
348  // For most user code, this is sufficient though, as it
349  // can not know the difference between executing loop()
350  // *in* start or *right after* start (the latter is
351  // guaranteed by the API).
352  // @see ActivityInterface::trigger for how trigger uses this
353  // assumption.
354  if ( rtos_sem_value(&sem) > 0 )
355  return true;
356 #endif
357  rtos_sem_signal(&sem);
358  return true;
359  }
360 
361  active=true;
362  if ( this->initialize() == false || active == false ) {
363  active = false;
364  return false;
365  }
366 
367  running = true;
368  rtos_sem_signal(&sem);
369 
370  return true;
371  }
372  else {
373 
374  if ( active )
375  return false;
376  active = true;
377 
378  bool result;
379  result = this->initialize();
380 
381  if (result == false || active == false) // detect call to stop() within initialize()
382  {
383  active = false;
384  return false;
385  }
386 
387  running = true;
388 
389  // signal start :
390  rtos_task_make_periodic(&rtos_task, period);
391  int ret = rtos_sem_signal(&sem);
392  if (ret != 0)
393  log(Critical)
394  << "Thread::start(): sem_signal returns " << ret
395  << endlog();
396  // do not wait, we did our job.
397 
398  return true;
399  }

416  bool Thread::stop()
417  {
418  if (!active)
419  return false;
420 
421  running = false;
422 
423  if ( period == 0)
424  {
425  if ( inloop ) {
426  if ( !this->breakLoop() ) {
427  log(Warning) << "Failed to stop thread " << this->getName() << ": breakLoop() returned false."<<endlog();
428  running = true;
429  return false;
430  }
431  // breakLoop was ok, wait for loop() to return.
432  }
433  // always take this lock, but after breakLoop was called !
434  MutexTimedLock lock(breaker, getStopTimeout());
435  if ( !lock.isSuccessful() ) {
436  log(Error) << "Failed to stop thread " << this->getName() << ": breakLoop() returned true, but loop() function did not return after " << getStopTimeout() <<" seconds."<<endlog();
437  running = true;
438  return false;
439  }
440  } else {
441  //
442  MutexTimedLock lock(breaker, getStopTimeout() );
443  if ( lock.isSuccessful() ) {
444  // drop out of periodic mode.
445  rtos_task_make_periodic(&rtos_task, 0);
446  } else {
447  log(Error) << "Failed to stop thread " << this->getName() << ": step() function did not return after "<< getStopTimeout() <<" seconds."<<endlog();
448  running = true;
449  return false;
450  }
451  }
452 
453  this->finalize();
454  active = false;
455  return true;
456  }

541  bool Thread::setPeriod(secs s, nsecs ns)
542  {
543  nsecs nsperiod = ns + 1000* 1000* 1000* s ;
544  if (nsperiod < 0)
545  return false;
546  // logic to switch from per->nper || nper->per
547  if ( (nsperiod == 0 && period != 0) || (nsperiod != 0 && period == 0)) {
548  // switch between periodic/non-periodic
549  // note for RTAI: the fosi_internal layer must detect if this is called from
550  // within rtos_task or outside the thread.
551  rtos_task_make_periodic(&rtos_task, nsperiod);
552  // jump from non periodic into periodic: first sample.
553  if ( period == 0) {
554  period = nsperiod; // avoid race with sem in thread func.
555  rtos_sem_signal(&sem);
556  }
557  }
558  // update rate:
559  period = nsperiod;
560 
561  return true;
562  }
