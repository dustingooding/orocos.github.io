157  static base::ChannelElement<T>* buildDataStorage(ConnPolicy const& policy, const T& initial_value = T())
158  {
159  if (policy.type == ConnPolicy::DATA)
160  {
161  typename base::DataObjectInterface<T>::shared_ptr data_object;
162  switch (policy.lock_policy)
163  {
164 #ifndef OROBLD_OS_NO_ASM
165  case ConnPolicy::LOCK_FREE:
166  data_object.reset( new base::DataObjectLockFree<T>(initial_value, policy) );
167  break;
168 #else
169  case ConnPolicy::LOCK_FREE:
170  RTT::log(Warning) << "lock free connection policy is unavailable on this system, defaulting to LOCKED" << RTT::endlog();
171 #endif
172  case ConnPolicy::LOCKED:
173  data_object.reset( new base::DataObjectLocked<T>(initial_value) );
174  break;
175  case ConnPolicy::UNSYNC:
176  data_object.reset( new base::DataObjectUnSync<T>(initial_value) );
177  break;
178  }
179  return new ChannelDataElement<T>(data_object, policy);
180  }
181  else if (policy.type == ConnPolicy::BUFFER || policy.type == ConnPolicy::CIRCULAR_BUFFER)
182  {
183  typename base::BufferInterface<T>::shared_ptr buffer_object;
184  switch (policy.lock_policy)
185  {
186 #ifndef OROBLD_OS_NO_ASM

189  break;

196  break;
197  case ConnPolicy::UNSYNC:
198  buffer_object.reset(new base::BufferUnSync<T>(policy.size, initial_value, policy));

220  {
221  typename internal::ConnInputEndpoint<T>::shared_ptr endpoint = port.getEndpoint();
222  typename base::ChannelElement<T>::shared_ptr buffer = port.getSharedBuffer();
223 
224  // Set buffer policy of the port
225  if (!(endpoint->setBufferPolicy(policy.buffer_policy))) {
226  log(Error) << "You mixed incompatible buffer policies for output port " << port.getName() << ": "
227  << "The new connection requests a " << policy.buffer_policy << " policy, "
228  << "but the port already has a " << endpoint->getBufferPolicy() << " policy." << endlog();
229  return typename internal::ConnOutputEndpoint<T>::shared_ptr();
230  }
231 
232  // Note: PerInputPort implies PUSH and PerOutputPort implies PULL
233  bool pull = policy.pull;
234  if (policy.buffer_policy == PerInputPort) pull = ConnPolicy::PUSH;
235  if (policy.buffer_policy == PerOutputPort) pull = ConnPolicy::PULL;
236 
237  if (pull == ConnPolicy::PULL && !force_unbuffered) {
238  if (!buffer) {
239  buffer = buildDataStorage<T>(policy, port.getLastWrittenValue());
240  if (!buffer) return typename internal::ConnOutputEndpoint<T>::shared_ptr();
241 
242  if (policy.buffer_policy == PerOutputPort) {
243  // For PerOutputPort connections, the buffer is installed BEFORE the input endpoint!
244  if (endpoint->connected()) {
245  log(Error) << "You tried to create a shared output buffer connection for output port " << port.getName() << ", "
246  << "but the port already has at least one incompatible outgoing connection." << endlog();
247  return typename internal::ConnOutputEndpoint<T>::shared_ptr();
248  }
249  return buffer->connectTo(endpoint) ? endpoint : typename internal::ConnInputEndpoint<T>::shared_ptr();
250  } else {
251  return endpoint->connectTo(buffer, policy.mandatory) ? buffer : typename internal::ConnInputEndpoint<T>::shared_ptr();
252  }
253

465  template<typename T>
466  static bool createConnection(OutputPort<T>& output_port, base::InputPortInterface& input_port, ConnPolicy const& policy)
467  {
468  if ( !output_port.isLocal() ) {
469  log(Error) << "Need a local OutputPort to create connections." <<endlog();
470  return false;
471  }
472 
473  if (output_port.connectedTo(&input_port)) {
474  log(Info) << "OutputPort " << output_port.getName() << " is already connected to " << input_port.getName() << ", ignoring new connection." << endlog();
475  return true;
476  }
477 
478  InputPort<T>* input_p = dynamic_cast<InputPort<T>*>(&input_port);
479 
480  // Shared push connection? => forward to createAndCheckSharedConnection()
481  if (policy.buffer_policy == Shared) {
482  return createAndCheckSharedConnection(&output_port, &input_port, buildSharedConnection<T>(&output_port, &input_port, policy), policy);
483  }
484 
485  // This is the input and output channel element of the output half
486  base::ChannelElementBase::shared_ptr output_half;
487  if (input_port.isLocal() && policy.transport == 0)
488  {
489  // Local connection
490  if (!input_p)
491  {
492  log(Error) << "Port " << input_port.getName() << " is not compatible with " << output_port.getName() << endlog();
493  return false;
494  }
495 
496  // local ports, create buffer here.
497  output_half = buildChannelOutput<T>(*input_p, policy, output_port.getLastWrittenValue());
498  }
499  else
500  {
501  // if the input is not local, this is a pure remote connection,
502  // if the input *is* local, the user requested to use a different transport
503  // than plain memory, rare case, but we accept it. The unit tests use this for example
504  // to test the OOB transports.
505  if ( !input_port.isLocal() ) {
506  output_half = buildRemoteChannelOutput( output_port, input_port, policy);
507  } else if (input_p) {
508  return createOutOfBandConnection<T>( output_port, *input_p, policy);
509  } else {
510  log(Error) << "Port " << input_port.getName() << " is not compatible with " << output_port.getName() << endlog();
511  return false;
512  }
513  }
514 
515  if (!output_half)
516  return false;
517 
518  // Since output is local, buildChannelInput is local as well.
519  // This this the input channel element of the whole connection
520  base::ChannelElementBase::shared_ptr channel_input;
521  channel_input = buildChannelInput<T>(output_port, policy);
522 
523  if (!channel_input) {
524  output_half->disconnect(true);
525  return false;
526  }
527 
528  // NOTE: channel_input and output_half are not yet connected!
529  return createAndCheckConnection(output_port, input_port, channel_input, output_half, policy);
530  }

585  // constructs an out-of-band channel:
586  // output_port -> channel_input -> stream_input -> (out-of-band transport of data) -> stream_output -> channel_output -> input_port
