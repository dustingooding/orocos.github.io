81  void StateMachine::loading() {
82  TRACE_INIT();
83  TRACE( "Being Loaded in ExecutionEngine." );
84  smStatus = Status::inactive;
85  for(TransitionMap::iterator it=stateMap.begin(); it != stateMap.end(); ++it) {
86  // inform all entry/exit state scripts:
87  if (it->first)
88  it->first->loaded( this->getEngine() );
89  // inform all transition scripts
90  for(TransList::iterator tlit= it->second.begin(); tlit != it->second.end(); ++tlit ) {
91  if ( get<4>(*tlit) )
92  get<4>(*tlit)->loaded( this->getEngine() );
93  }
94  }
95  for( EventMap::iterator it = eventMap.begin(); it != eventMap.end(); ++it)
96  {
97  // inform all transition scripts
98  for(EventList::iterator tlit= it->second.begin(); tlit != it->second.end(); ++tlit ) {
99  if ( get<5>(*tlit) )
100  get<5>(*tlit)->loaded( this->getEngine() );
101  }
102  }

105  void StateMachine::unloading() {
106  TRACE_INIT();
107  TRACE( "Being unloaded from ExecutionEngine." );
108  if ( this->isActive() == false)
109  return;
110  if ( this->isReactive() )
111  this->requestFinalState();
112  if ( this->isAutomatic() )
113  this->stop();
114  if (this->currentState() != this->getFinalState() )
115  this->execute(); // try one last time
116  if (this->currentState() != this->getFinalState() )
117  log(Critical) << "Failed to bring StateMachine "<< this->getName()
118  << " into the final state. Program stalled in state '"
119  << this->currentState()->getName()<<"' line number "
120  << this->getLineNumber()<<endlog(); // critical failure !

263  bool StateMachine::execute()
264  {
265  TRACE_INIT();
266 
267  // before dealing with transitional states,
268  // check if we're actually running.
269  if (smStatus == Status::inactive || smStatus == Status::unloaded) {
270  smpStatus = nill;
271  return true;
272  }
273  os::MutexLock lock(execlock);
274 
275  // internal transitional issues.
276  switch (smpStatus) {
277  case pausing:
278  TRACE("Is paused now.");
279  smStatus = Status::paused;
280  smpStatus = nill;
281  return true;
282  break;
283  case gostop:
284  this->executePending();
285  if ( this->requestFinalState() ) {
286  // test for atomicity :
287  if ( this->inTransition() ) {
288  smStatus = Status::stopping;
289  } else {
290  TRACE("Is stopped now.");
291  smStatus = Status::stopped;
292  }
293  smpStatus = nill;
294  }
295  return true;
296  break;
297  case goreset:
298  if ( this->executePending() ) {
299  this->requestInitialState();
300  if ( this->inTransition() ) {
301  smStatus = Status::resetting;
302  } else {
303  TRACE("Is reset now.");
304  smStatus = Status::active;
305  }
306  }
307  smpStatus = nill;
308  return true;
309  break;
310  case nill:
311  break;
312  }
313 
314  // public visible states.
315  switch (smStatus) {
316  case Status::inactive:
317  return true;
318  break;
319  case Status::requesting:
320  if ( this->executePending() ) { // if all steps done,
321  this->requestNextState();
322  this->executePending(); // execute steps of next state
323  TRACE("Is active now.");
324  smStatus = Status::active;
325  }
326  break;
327  case Status::active:
328  this->executePending();
329  break;
330  case Status::running:
331  if ( this->executePending() == false) {
332  TRACE("Yielding...");
333  break;
334  }
335  // if all pending done:
336  this->requestNextState(); // one state at a time
337  this->executePending(); // execute steps of next state
338  break;
339  case Status::paused:
340  if (mstep) {
341  if ( this->executePending(true) ) { // if all steps done,
342  this->requestNextState(true); // one state at a time
343  this->executePending(true); // execute steps of next state
344  }
345  TRACE("Did a step.");
346  mstep = false;
347  }
348  break;
349  case Status::error:
350  case Status::unloaded:
351  break;
352  case Status::activating:
353  this->executePending();
354  if ( !this->inTransition() ) {
355  TRACE("Is active now.");
356  smStatus = Status::active;
357  }
358  break;
359  case Status::stopping:
360  if ( this->executePending() ) {
361  TRACE("Is stopped now.");
362  smStatus = Status::stopped;
363  }break;
364  case Status::deactivating:
365  if ( this->executePending() ) {
366  TRACE("Is inactive now.");
367  smStatus = Status::inactive;
368  }break;
369  case Status::resetting:
370  if ( this->executePending() ) {
371  TRACE("Is reset now.");
372  smStatus = Status::active;
373  }break;
374  case Status::stopped:
375  this->executePending();
376  if ( current != finistate ) {// detect leaving final state by event/request
377  smStatus = Status::active;
378  } break;
379  }
380  return true;
381  }

400  {
401  TRACE_INIT();
402  os::MutexLock lock(execlock);
403  // if we are inactive or in transition, don't do anything.
404  if ( current == 0 || ( !inError() && !interruptible() ) ) {

418  void StateMachine::changeState(StateInterface* newState, ProgramInterface* transProg, bool stepping) {
419  TRACE_INIT();
420  if ( newState == current )
421  {
422  // this is only true if current state was selected in a transition of current.
423  if ( transProg ) {
424  TRACE("Transition triggered to self: '"+current->getName()+"'");
425  transProg->reset();
426  if (transProg->start() == false )
427  smStatus = Status::error;
428  currentTrans = transProg;
429  // manually reset reqstep, or the next iteration would skip transition checks.
430  reqstep = stateMap.find( current )->second.begin();
431  // from now on, we are in transition to self !
432  // currentRun is _not_ set to zero or reset.
433  // it is/may be interrupted by trans, then continued.
434  // if already in handle, it is completed before trans, then trans is executed.
435  // since a transProg was given, no handle is executed.
436  } else {
437  // execute the default action (schedule handle )
438  // if no transition to another state took place and no transprog specified.
439  // only schedule a handle if not yet in progress.
440  if ( currentHandle == 0 )
441  handleState( current );
442  }
443  }
444  else
445  {
446  TRACE("Transition triggered from '"+ (current ? current->getName() : "null") +"' to '"+(newState ? newState->getName() : "null")+"'.");
447  // reset handle and run, in case it is still set ( during error
448  // or when an event arrived ).
449  currentRun = 0;
450  currentHandle = 0;
451  if ( transProg ) {
452  transProg->reset();
453  if ( transProg->start() == false )
454  smStatus = Status::error;
455 
456  }
457  currentTrans = transProg;
458 
459  next = newState;
460 
461  disableEvents(current);
462  leaveState(current);
463  assert( currentEntry == 0); // we assume that in executePending, trans and exit get executed first and entry is stil null.
464  }
465  // schedule a run for the next 'step'.
466  // if handle above finished, run will be called directly
467  // in executePending. if handle was not finished
468  // or stepping, it will be called after handle.
469  runState( newState );
470  }

512  bool StateMachine::createEventTransition( ServicePtr sp, ExecutionEngine* target_engine,
513  const std::string& ename, vector<DataSourceBase::shared_ptr> args,
514  StateInterface* from, StateInterface* to,
515  ConditionInterface* guard, boost::shared_ptr<ProgramInterface> transprog,
516  StateInterface* elseto, boost::shared_ptr<ProgramInterface> elseprog )
517  {
518  Logger::In in("StateMachine::createEventTransition");
519  DisposableInterface::shared_ptr di = sp->getLocalOperation(ename);
520  OperationCallerInterface::shared_ptr oci = dynamic_pointer_cast<OperationCallerInterface>(di);
521  if ( !oci ) {
522  log(Error) << "Can not receive event '"<< ename <<"' in StateMachine : not a local operation."<< endlog();
523  return false;
524  }
525 
526  if ( !( sp && guard ) ) {
527  log(Error) << "Invalid arguments for event '"<< ename <<"'. ";
528  if (!sp)
529  log() <<"Service was null. ";
530  if (!guard)
531  log() <<"Guard Condition was null. ";
532  log() << endlog();
533  return false;
534  }
535 
536  if ( to == 0 )
537  to = from;
538 
539  // get ename from event service, provide args as arguments
540  // event should be activated upon entry of 'from'
541  // guard is evaluated to get final 'ok'.
542  // event does a requestStateChange( to );
543  // if 'ok', execute transprog during transition.
544 
545  // Store guard and transprog for copy/clone semantics.
546  // upon SM copy, recreate the handles for the copy SM with a copy of guard/transprog.
547  // Same for args. I guess we need to store all arguments of this function to allow
548  // proper copy semantics, such that the event handle can be created for each new SM
549  // instance. Ownership of guard and transprog is to be determined, but seems to ly
550  // with the SM. handle.destroy() can be called upon SM destruction.
551  Handle handle;
552 
553  log(Debug) << "Creating Signal handler for Operation '"<< ename <<"' from state "<< (from ? from->getName() : string("(global)")) << " to state " << ( to ? to->getName() : string("(global)") ) <<Logger::endl;
554 #ifdef ORO_SIGNALLING_OPERATIONS
555  // don't deliver this signal asynchronously, we want it right away.
556  handle = sp->produceSignal( ename, new CommandFunction( boost::bind( &StateMachine::eventTransition, this, from, guard, transprog.get(), to, elseprog.get(), elseto) ), args, 0 );
557 #endif
558  if ( !handle.ready() ) {
559  Logger::log() << Logger::Error << "Could not setup handle for event '"<<ename<<"'."<<Logger::endl;
560  return false; // event does not exist...
561  }
562  // all our handles must start in disconnected state:
563  handle.disconnect();
564  // BIG NOTE : we MUST store handle otherwise, the connection is destroyed (cfr setup vs connect).
565  // Off course, we also need it to connect/disconnect the event.
566  eventMap[from].push_back( boost::make_tuple( sp, ename, args, to, guard, transprog, handle, elseto, elseprog) );
567  // add the states to the statemap.
568  stateMap[from];
569  stateMap[to];
570  return true;
571  }

573  bool StateMachine::eventTransition(StateInterface* from, ConditionInterface* c, ProgramInterface* p, StateInterface* to, ProgramInterface* elsep, StateInterface* elseto )
574  {
575  TRACE_INIT();
576  // called by event to begin Transition to 'to'.
577  // This interrupts the current run program at yield point
578  // the transition and/or exit program can cleanup...
579 
580  MutexLock lock(execlock); // recursive
581 
582  if ( !current)
583  return true;
584 
585  // Only transition if this event was meant for this state and we are not
586  // in transition already.
587  // If condition fails, check precondition 'else' state (if present) and
588  // execute else program (may be null).
589 
590  if (from == 0)
591  from = current;
592  if (to == 0)
593  to = current;
594  TRACE("Received Signal in state '"+ current->getName()+"' for transition from state '" + from->getName() + "' to state '" + to->getName() + "'");
595  if ( from == current && !this->inTransition() ) {
596  if ( c->evaluate() && checkConditions(to, false) == 1 ) {
597  TRACE( "Valid transition from " + from->getName() +
598  +" to "+to->getName()+".");
599  // stepping == false ! This executes the whole state transition
600  changeState( to, p, false ); // valid transition to 'to'. Q: should stepping be on or off ?
601  // trigger EE in order to force execution of the remainder of the state transition:
602  this->getEngine()->getActivity()->timeout();
603  }
604  else {
605  if ( c->evaluate() == false )
606  TRACE( "Rejected transition from " + from->getName() +
607  " to " + to->getName() +
608  " within state " + current->getName() + ": guards failed.");
609  if ( checkConditions(to,false) != 1 )
610  TRACE( "Rejected transition from " + from->getName() +
611  " to " + to->getName() +
612  " within state " + current->getName() + ": preconditions failed.");
613  }
614  }
615 #if 1
616  else {
617  if (this->inTransition() ) {
618  TRACE( "Rejected transition from " + from->getName() +
619  " within " + current->getName() + ": already in transition.");
620  } else {
621  TRACE( "Rejected transition from " + from->getName() +
622  + " within " + current->getName() + ": wrong state.");
623  }
624  }
625 #endif
626  return true;
627  }

629  StateInterface* StateMachine::requestNextState(bool stepping)
630  {
631  TRACE_INIT();
632  os::MutexLock lock(execlock);
633  // bad idea, user, don't run this if we're not active...
634  if( current == 0 )
635  return 0;
636  // only a run program may be interrupted...
637  if ( !interruptible() || currentTrans ) {
638  return current; // can not accept request, still in transition.
639  }
640 
641  // Reset global conditions.
642  TransList::const_iterator it, it1, it2;
643  it1 = stateMap.find( 0 )->second.begin();
644  it2 = stateMap.find( 0 )->second.end();
645 
646  if ( reqstep == stateMap.find( current )->second.begin() ) // avoid reseting too much in stepping mode.
647  for ( it= it1; it != it2; ++it)
648  get<0>(*it)->reset();
649 
650  if ( reqstep == reqend ) { // if nothing to evaluate, eval globals, then just handle()
651 
652  for ( ; it1 != it2; ++it1 )
653  if ( get<0>(*it1)->evaluate()
654  && checkConditions( get<1>(*it1) ) == 1 ) {
655  StateInterface* next = get<1>(*it1);
656  if ( next == 0 ) // handle current if no next
657  changeState( current, get<4>(*it1).get(), stepping );
658  else
659  changeState( next, get<4>(*it1).get(), stepping );
660  // the request was accepted
661  return current;
662  }
663  // no transition found: handle()
664  changeState( current, 0, stepping );
665  return current;
666  }
667 
668  // if we got here, at least one evaluation to check
669  do {
670  if ( get<0>(*reqstep)->evaluate() ) {
671  // evaluate() might call stop() or other sm functions:
672  if (reqstep == reqend )
673  return current;
674  // check preconds of target state :
675  int cres = checkConditions( get<1>(*reqstep), stepping );
676  if (cres == 0) {
677  break; // only returned in stepping
678  }
679  if( cres == 1) {
680  changeState( get<1>(*reqstep), get<4>(*reqstep).get(), stepping );
681  break; // valid transition
682  }
683  // if cres == -1 : precondition failed, increment reqstep...
684  }
685  if ( reqstep + 1 == reqend ) {
686  // to a state specified by the user (global)
687  for ( ; it1 != it2; ++it1 ) {
688  if ( get<0>(*it1)->evaluate() && checkConditions( get<1>(*it1) ) == 1 ) {
689  StateInterface* next = get<1>(*it1);
690  if ( next == 0) // handle current if no next
691  changeState( current, get<4>(*it1).get(), stepping );
692  else
693  changeState( next, get<4>(*it1).get(), stepping );
694  // the request was accepted
695  return current;
696  }
697  }
698  // no transition was found, reset and 'schedule' a handle :
699  reqstep = stateMap.find( current )->second.begin();

713  int StateMachine::checkConditions( StateInterface* state, bool stepping ) {
714 
715  // if the preconditions of \a state are checked the first time in stepping mode, reset the iterators.
716  if ( !checking_precond || !stepping ) {
717  prec_it = precondMap.equal_range(state); // state is the _target_ state
718  }
719 
720  // will be set to true if stepping below.
721  //checking_precond = false;
722 
723  while ( prec_it.first != prec_it.second ) {
724  if (checking_precond == false && stepping ) {
725  evaluating = prec_it.first->second.second; // indicate we will evaluate this line (if any).
726  checking_precond = true;
727  return 0;
728  }
729  if ( prec_it.first->second.first->evaluate() == false ) {
730  checking_precond = false;
731  return -1; // precondition failed
732  }
733  ++( prec_it.first );
734  if (stepping) {
735  if ( prec_it.first != prec_it.second )
736  evaluating = prec_it.first->second.second; // indicate we will evaluate the next line (if any).
737  checking_precond = true;
738  return 0; // not done yet.
739  }
740  }
741  checking_precond = false;
742  return 1; // success !
743  }

800  bool StateMachine::requestStateChange( StateInterface * s_n )
801  {
802  os::MutexLock lock(execlock);
803  // bad idea, user, don't run this if we're not active...
804  if( current == 0 )
805  return false;
806 
807  if ( !interruptible() ) {
808  return false; // can not accept request, still in transition
809  }
810 
811  // we may make transition to next state :
812 
813  // to current state
814  if ( current == s_n )
815  {
816  changeState( s_n, 0 );
817  executePending();
818  return true;
819  }
820 
821  // between 2 states specified by the user.
822  TransList::iterator it, it1, it2;
823  it1 = stateMap.find( current )->second.begin();
824  it2 = stateMap.find( current )->second.end();
825 
826  for ( ; it1 != it2; ++it1 )
827  if ( get<1>(*it1) == s_n
828  && get<0>(*it1)->evaluate()
829  && checkConditions( s_n ) == 1 ) {
830  changeState( s_n, get<4>(*it1).get() );
831  // the request was accepted
832  executePending();
833  return true;
834  }
835 
836  // to a state specified by the user (global)
837  it1 = stateMap.find( 0 )->second.begin();
838  it2 = stateMap.find( 0 )->second.end();
839 
840  // reset all conditions
841  for ( it= it1; it != it2; ++it)
842  get<0>(*it)->reset();
843 
844  // evaluate them
845  for ( ; it1 != it2; ++it1 )
846  if ( get<1>(*it1) == s_n
847  && get<0>(*it1)->evaluate()
848  && checkConditions( s_n ) == 1 ) {
849  changeState( s_n, get<4>(*it1).get() );
850  // the request was accepted
851  executePending();
852  return true;
853  }
854 
855  // to final state
856  if ( finistate == s_n )
857  {
858  changeState( s_n, 0 );
859  executePending();
860  return true;
861  }
862 
863  // to inital state from final state
864  if ( initstate == s_n && current == finistate)
865  {
866  changeState( s_n, 0 );
867  executePending();
868  return true;
869  }
870 
871  // the request has failed.
872  return false;
873  }

997  bool StateMachine::executePending( bool stepping )
998  {
999  TRACE_INIT();
1000  // This function has great resposibility, since it acts like
1001  // a scheduler for pending requests. It tries to devise what to
1002  // do on basis of the contents of variables (like current*, next,...).
1003  // This is a somewhat
1004  // fragile implementation but requires very little bookkeeping.
1005  // if returns true : a transition (exit/entry) is done
1006  // and a new state may be requested.
1007 
1008  if ( inError() )
1009  return false;
1010 
1011  // first try to finish the current entry program, if any:
1012  if ( currentEntry ) {
1013  TRACE("Executing entry program of '"+ (current ? current->getName() : "(null)") +"'" );
1014  if ( this->executeProgram(currentEntry, stepping) == false )
1015  return false;
1016  // done.
1017  TRACE("Finished entry program of '"+ (current ? current->getName() : "(null)") +"'" );
1018  // in stepping mode, delay 'true' one executePending().
1019  if ( stepping ) {
1020  currentProg = currentRun;
1021  return false;
1022  }
1023  }
1024 
1025  // if a transition has been scheduled, proceed directly instead of doing a run:
1026  if ( currentTrans ) {
1027  TRACE("Executing transition program from '"+ (current ? current->getName() : "(null)") + "' to '"+ ( next ? next->getName() : "(null)")+"'" );
1028  // exception : transition during handle, first finish handle !
1029  if ( currentHandle ) {
1030  if ( this->executeProgram(currentHandle, stepping) == false )
1031  return false;
1032  } else
1033  if ( this->executeProgram(currentTrans, stepping) == false )
1034  return false;
1035  // done.
1036  TRACE("Finished transition program from '"+ (current ? current->getName() : "(null)") + "' to '"+ ( next ? next->getName() : "(null)")+"'" );
1037  // in stepping mode, delay 'true' one executePending().
1038  if ( stepping ) {
1039  currentProg = currentExit ? currentExit : (currentEntry ? currentEntry : currentRun);
1040  return false;
1041  }
1042  }
1043 
1044  // last is exit
1045  if ( currentExit ) {
1046  TRACE("Executing exit program from '"+ (current ? current->getName() : "(null)") + "' (going to '"+ (next ? next->getName() : "(null)") +"')" );
1047  if ( this->executeProgram(currentExit, stepping) == false )
1048  return false;
1049  // done.
1050  TRACE("Finished exit program from '"+ (current ? current->getName() : "(null)") + "' (going to '"+ (next ? next->getName() : "(null)") +"')" );
1051  // in stepping mode, delay 'true' one executePending().
1052  if ( stepping ) {
1053  currentProg = (currentEntry ? currentEntry : currentRun);
1054  return false;
1055  }
1056  }
1057 
1058  // only reset the reqstep if we changed state.
1059  // if we did not change state, it will be reset in requestNextState().
1060  if ( current != next ) {
1061  if ( next ) {
1062  reqstep = stateMap.find( next )->second.begin();
1063  reqend = stateMap.find( next )->second.end();
1064  // init for getLineNumber() :
1065  if ( reqstep == reqend )
1066  evaluating = 0;
1067  else
1068  evaluating = get<3>(*reqstep);
1069  } else {
1070  current = 0;
1071  return true; // done if current == 0 !
1072  }
1073  // make change transition after exit of previous state:
1074  TRACE("Formally transitioning from '"+ (current ? current->getName() : "(null)") + "' to '"+ (next ? next->getName() : "(null)") +"'" );
1075 
1076  current = next;
1077  enterState(next);
1078  enableEvents(next);
1079  }
1080 
1081  // finally, execute the current Entry of the new state:
1082  if ( currentEntry ) {
1083  TRACE("Executing entry program of '"+ (current ? current->getName() : "(null)") +"'" );
1084  if ( this->executeProgram(currentEntry, stepping) == false )
1085  return false;
1086  // done.
1087  TRACE("Finished entry program of '"+ (current ? current->getName() : "(null)") +"'" );
1088  // in stepping mode, delay 'true' one executePending().
1089  if ( stepping ) {
1090  currentProg = currentRun;
1091  return false;
1092  }
1093  // Either way, if currentEntry was set here, we get out by returning true or false.
1094  // currentHandle is unlikely to be set, but currentRun is set by 'changeState' and
1095  // should only be executed the next execution cycle (if no transition is scheduled by then).
1096  return true;
1097  }
1098 
1099  // Handle is executed after the transitions failed.
1100  if ( currentHandle ) {
1101  TRACE("Executing handle program of '"+ (current ? current->getName() : "(null)") +"'" );
1102  if ( this->executeProgram(currentHandle, stepping) == false )
1103  return false;
1104  // done.
1105  TRACE("Finished handle program of '"+ (current ? current->getName() : "(null)") +"'" );
1106  // in stepping mode, delay 'true' one executePending().
1107  if ( stepping ) {
1108  currentProg = currentRun;
1109  return false;
1110  }
1111  }
1112 
1113  // Run is executed before the transitions.
1114  if ( currentRun ) {
1115  TRACE("Executing run program of '"+ (current ? current->getName() : "(null)") +"'" );
1116  if ( this->executeProgram(currentRun, stepping) == false )
1117  return false;
1118  // done.
1119  TRACE("Finished run program of '"+ (current ? current->getName() : "(null)") +"'" );
1120  // in stepping mode, delay 'true' one executePending().
1121  if ( stepping )
1122  return false;
1123  }
1124 
1125  return true; // all pending is done
1126  }

1234  {

1237  if ( smStatus == Status::inactive) {
