49  bool PropertyBagVisitor::introspectAndDecompose(PropertyBase* v)
50  {
51  DataSourceBase::shared_ptr dsb = v->getTypeInfo()->decomposeType( v->getDataSource() );
52  // Try conversion first because this is the user's implementation of decomposition:
53  if ( dsb ) {
54  // check if type returned itself to avoid decomposition:
55  if ( dsb == v->getDataSource() )
56  return false;
57  base::PropertyBase* p = dsb->getTypeInfo()->buildProperty(v->getName(), v->getDescription(), dsb);
58  this->introspect( p );
59  delete p;
60  return true;
61  } else {
62  // now try generic decomposition, based on getMember():
63  Property<PropertyBag> res(v->getName(), v->getDescription() );
64  if ( types::propertyDecomposition(v, res.value() ) ) {
65  this->introspect( res );
66  deletePropertyBag( res.value() );
67  return true;
68  }
69  // Cleanup.
70  deletePropertyBag( res.value() );
71  // All failed.
72  return false;
73  }
74  return false;
75  }
