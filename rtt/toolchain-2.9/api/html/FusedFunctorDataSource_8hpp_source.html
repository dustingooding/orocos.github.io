294  bool evaluate() const {
295  // put the member's object as first since SequenceFactory does not know about the OperationCallerBase type.
296  typedef bf::cons<base::OperationCallerBase<Signature>*, typename SequenceFactory::data_type> arg_type;
297  typedef typename AddMember<Signature,base::OperationCallerBase<Signature>* >::type call_type;
298  // this foo pointer dance is because older compilers don't handle using
299  // &bf::invoke<call_type,arg_type> directly.
300  typedef typename bf::result_of::invoke<call_type,arg_type>::type iret;
301  typedef iret(*IType)(call_type, arg_type const&);
302  IType foo = &bf::invoke<call_type,arg_type>;
303  // we need to store the ret value ourselves.
304  ret.exec( boost::bind(foo, &base::OperationCallerBase<Signature>::call, arg_type(ff.get(), SequenceFactory::data(args))) );
305  if(ret.isError()) {
306  ff->reportError();
307  ret.checkError();
308  }
309  SequenceFactory::update(args);
310  return true;
311  }

559  if ( subscriber ) {
