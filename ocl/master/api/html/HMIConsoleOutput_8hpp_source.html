86  void updateHook()
87  {
88  {
89  RTT::os::MutexLock lock1( msg_lock );
90  if ( ! messages.str().empty() ) {
91  std::cout << coloron << _prompt<< coloroff <<
92  messages.str() << std::endl;
93  messages.rdbuf()->str("");

95  }

98  if ( ! logmessages.str().empty() ) {
