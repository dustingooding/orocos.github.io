69  INTERNAL_QUAL int rtos_task_create(RTOS_TASK* task,
70  int priority,
71  unsigned cpu_affinity,
72  const char * name,
73  int sched_type,
74  size_t stack_size,
75  void * (*start_routine)(void *),
76  ThreadInterface* obj)
77  {
78  int rv; // return value
79  rtos_task_check_priority( &sched_type, &priority );
80  // Save priority internally, since the pthread_attr* calls are broken !
81  // we will pick it up later in rtos_task_set_scheduler().
82  task->priority = priority;
83  task->wait_policy = ORO_WAIT_ABS;
84 
85  // Set name
86  if ( strlen(name) == 0 )
87  name = "Thread";
88  task->name = strcpy( (char*)malloc( (strlen(name) + 1) * sizeof(char)), name);
89 
90  if ( (rv = pthread_attr_init(&(task->attr))) != 0 ){
91  return rv;
92  }
93  // Set scheduler type (_before_ assigning priorities!)
94  if ( (rv = pthread_attr_setschedpolicy(&(task->attr), sched_type)) != 0){
95  return rv;
96  }
97  pthread_attr_getschedpolicy(&(task->attr), &rv );
98  assert( rv == sched_type );
99 
100  struct sched_param sp;
101  sp.sched_priority=priority;
102  // Set priority
103  if ( (rv = pthread_attr_setschedparam(&(task->attr), &sp)) != 0 ){
104  return rv;
105  }
106  rv = pthread_create(&(task->thread), &(task->attr), start_routine, obj);
107  log(Debug) <<"Created Posix thread "<< task->thread <<endlog();
108  return rv;
109  }

126  INTERNAL_QUAL int rtos_task_set_scheduler(RTOS_TASK* task, int sched_type)
127  {
128  int policy = -1;
129  struct sched_param param;
130  // first check the argument
131  if ( task && task->thread != 0 && rtos_task_check_scheduler( &sched_type) == -1 )
132  return -1;
133  // if sched_type is different, the priority must change as well.
134  if (pthread_getschedparam(task->thread, &policy, &param) == 0) {
135  // now update the priority
136  param.sched_priority = task->priority;
137  rtos_task_check_priority( &sched_type, &param.sched_priority );
138  // write new policy:
139  return pthread_setschedparam( task->thread, sched_type, &param);
140  }
141  return -1;
142  }

173  INTERNAL_QUAL int rtos_task_wait_period( RTOS_TASK* task )
174  {
175  if ( task->period == 0 )
176  return 0;
177 
178  //rtos_printf("Time is %lld nsec, Mark is %lld nsec.\n",rtos_get_time_ns(), task->periodMark );
179  // CALCULATE in nsecs
180  NANO_TIME timeRemaining = task->periodMark - rtos_get_time_ns();
181 
182  // next wake-up time :
183  if (task->wait_policy == ORO_WAIT_ABS)
184  {
185  task->periodMark += task->period;
186  }
187  else
188  {
189  task->periodMark = rtos_get_time_ns() + task->period;
190  }
191 
192  if ( timeRemaining > 0 ) {
193  //rtos_printf("Waiting for %lld nsec\n",timeRemaining);
194  TIME_SPEC ts( ticks2timespec( timeRemaining ) );
195  rtos_nanosleep( &ts , NULL );
196  return 0;
197  }
198 
199  return -1;
200  }
