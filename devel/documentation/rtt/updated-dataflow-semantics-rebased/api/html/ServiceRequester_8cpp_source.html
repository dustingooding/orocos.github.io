136  bool ServiceRequester::connectTo( Service::shared_ptr sp) {
137  for (OperationCallers::iterator it = mmethods.begin(); it != mmethods.end(); ++it) {
138  if ( !it->second->ready() ) {
139  if (sp->hasOperation( it->first )) {
140  it->second->setImplementation( sp->getLocalOperation( it->first ), mrowner ? mrowner->engine() : 0 );
141  if ( it->second->ready() ) {
142  log(Debug) << "Successfully set up OperationCaller " << it->first <<endlog();
143  if (!mrowner)
144  log(Debug) << "OperationCaller "<< it->first << " has no caller set: using GlobalEngine."<<endlog();
145  }
146  }
147  if (sp->hasMember( it->first )) {
148  it->second->setImplementationPart( sp->getOperation( it->first ), mrowner ? mrowner->engine() : 0 );
149  if ( it->second->ready() ) {
150  log(Debug) << "Successfully set up OperationCaller " << it->first <<endlog();
151  if (!mrowner)
152  log(Debug) << "OperationCaller "<< it->first << " has no caller set: using GlobalEngine."<<endlog();
153  }
154  }
155  }
156  }
157  if (ready()) {
158  if (!mprovider)
159  mprovider = sp;
160  log(Info) << "Found complete interface of requested service '" << mrname <<"'"<< endlog();
161  return true;
162  }
163 
164  return false;
165  }
