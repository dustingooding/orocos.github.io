257 bool MultipleInputsChannelElementBase::disconnect(ChannelElementBase::shared_ptr const& channel, bool forward)
258 {
259  if (channel) {
260  bool was_last = false;
261  {
262  // Remove the channel from the inputs list
263  RTT::os::MutexLock lock(inputs_lock);
264  Inputs::iterator found = std::find(inputs.begin(), inputs.end(), channel);
265  if (found == inputs.end()) {
266  return false;
267  }
268  ChannelElementBase::shared_ptr input = *found;
269 
270  if (!forward) {
271  if (!input->disconnect(this, forward)) {
272  return false;
273  }
274  }
275 
276  removeInput(input.get()); // invalidates input
277  was_last = inputs.empty();
278  }
279 
280  // If the removed input was the last channel and forward is true, disconnect output side, too.
281  if (was_last && forward) {
282  disconnect(true);
283  }
284 
285  return true;
286 
287  } else if (!forward) {
288  // Disconnect and remove all inputs
289  RTT::os::MutexLock lock(inputs_lock);
290  for (Inputs::iterator it = inputs.begin(); it != inputs.end(); ) {
291  const ChannelElementBase::shared_ptr &input = *it++;
292  input->disconnect(this, false);
293  removeInput(input.get()); // invalidates input
294  }
295  assert(inputs.empty());
296  }
297 
298  return ChannelElementBase::disconnect(channel, forward);
299 }
