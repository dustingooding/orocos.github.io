291  bool evaluate() const {
292  // put the member's object as first since SequenceFactory does not know about the OperationCallerBase type.
293  typedef bf::cons<base::OperationCallerBase<Signature>*, typename SequenceFactory::data_type> arg_type;
294  typedef typename AddMember<Signature,base::OperationCallerBase<Signature>* >::type call_type;
295  // this foo pointer dance is because older compilers don't handle using
296  // &bf::invoke<call_type,arg_type> directly.
297  typedef typename bf::result_of::invoke<call_type,arg_type>::type iret;
298  typedef iret(*IType)(call_type, arg_type const&);
299  IType foo = &bf::invoke<call_type,arg_type>;
300  // we need to store the ret value ourselves.
301  ret.exec( boost::bind(foo, &base::OperationCallerBase<Signature>::call, arg_type(ff.get(), SequenceFactory::data(args))) );
302  if(ret.isError()) {
303  ff->reportError();
304  ret.checkError();
305  }
306  SequenceFactory::update(args);
307  return true;
308  }
