239  void load_binary(void *address, std::size_t count)
240  {
241  // note: an optimizer should eliminate the following for char files
242  std::streamsize s = count / sizeof(Elem);
243  std::streamsize scount = m_sb.sgetn(
244  static_cast<Elem *> (address), s);
245  if (scount != static_cast<std::streamsize> (s))
246 #if BOOST_VERSION >= 104400
247  boost::serialization::throw_exception(
248  boost::archive::archive_exception(
249  boost::archive::archive_exception::input_stream_error));
250 #else
251  boost::serialization::throw_exception(
252  boost::archive::archive_exception(
253  boost::archive::archive_exception::stream_error));
254 #endif
255  // note: an optimizer should eliminate the following for char files
256  s = count % sizeof(Elem);
257  if (0 < s)
258  {
259  // if(is.fail())
260  // boost::serialization::throw_exception(
261  // archive_exception(archive_exception::stream_error)
262  // );
263  Elem t;
264  scount = m_sb.sgetn(&t, 1);
265  if (scount != 1)
266 #if BOOST_VERSION >= 104400
267  boost::serialization::throw_exception(
268  boost::archive::archive_exception(
269  boost::archive::archive_exception::input_stream_error));
270 #else
271  boost::serialization::throw_exception(
272  boost::archive::archive_exception(
273  boost::archive::archive_exception::stream_error));
274 #endif
275  std::memcpy(static_cast<char*> (address) + (count - s), &t,
276  s);
277  }
278  data_read += count;
279  }
