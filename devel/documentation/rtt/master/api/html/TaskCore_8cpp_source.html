104  bool TaskCore::configure() {
105  if ( mTaskState == Stopped || mTaskState == PreOperational) {
106  TRY(
107  mTargetState = Stopped;
108  if (configureHook() ) {
109  mTaskState = Stopped;
110  return true;
111  } else {
112  mTargetState = mTaskState = PreOperational;
113  return false;
114  }
115  ) CATCH(std::exception const& e,
116  log(Error) << "in configure(): switching to exception state because of unhandled exception" << endlog();
117  log(Error) << " " << e.what() << endlog();
118  exception();
119  ) CATCH_ALL(
120  log(Error) << "in configure(): switching to exception state because of unhandled exception" << endlog();
121  exception();
122  )
123  }
124  return false; // no configure when running.
125  }
126 
127  bool TaskCore::cleanup() {
128  if ( mTaskState == Stopped ) {
129  TRY(
130  mTargetState = PreOperational;
131  cleanupHook();
132  mTaskState = PreOperational;
133  return true;
134  ) CATCH(std::exception const& e,
135  log(Error) << "in cleanup(): switching to exception state because of unhandled exception" << endlog();
136  log(Error) << " " << e.what() << endlog();
137  exception();
138  ) CATCH_ALL (
139  log(Error) << "in cleanup(): switching to exception state because of unhandled exception" << endlog();
140  exception();
141  )
142  }
143  return false; // no cleanup when running or not configured.
144  }

159  void TaskCore::exception() {
160  //log(Error) <<"Exception happend in TaskCore."<<endlog();
161  TaskState copy = mTaskState;
162  mTargetState = mTaskState = Exception;
163  TRY (
164  if ( copy >= Running ) {
165  stopHook();
166  }
167  if ( copy >= Stopped && mInitialState == PreOperational ) {
168  cleanupHook();
169  }
170  exceptionHook();
171  ) CATCH(std::exception const& e,
172  log(RTT::Error) << "stopHook(), cleanupHook() and/or exceptionHook() raised " << e.what() << ", going into Fatal" << endlog();
173  fatal();
174  ) CATCH_ALL (
175  log(Error) << "stopHook(), cleanupHook() and/or exceptionHook() raised an exception, going into Fatal" << endlog();
176  fatal();
177  )
178  }

192  bool TaskCore::start() {
193  if ( mTaskState == Stopped ) {
194  TRY (
195  mTargetState = Running;

197  mTaskState = Running;

203  log(Error) << "in start(): switching to exception state because of unhandled exception" << endlog();
204  log(Error) << " " << e.what() << endlog();
205  exception();
206  ) CATCH_ALL (

208  exception();
209  )
210  }
211  return false;

214  bool TaskCore::stop() {
215  TaskState orig = mTaskState;
216  if ( mTaskState >= Running ) {
217  TRY(
218  mTargetState = Stopped;
219  if ( engine()->stopTask(this) ) {
220  stopHook();
221  mTaskState = Stopped;
222  return true;
223  } else {
224  mTaskState = orig;
225  mTargetState = orig;
226  }
227  ) CATCH(std::exception const& e,
228  log(Error) << "in stop(): switching to exception state because of unhandled exception" << endlog();
229  log(Error) << " " << e.what() << endlog();
230  exception();
231  ) CATCH_ALL (
232  log(Error) << "in stop(): switching to exception state because of unhandled exception" << endlog();
233  exception();
234  )
235  }
236  return false;
237  }
