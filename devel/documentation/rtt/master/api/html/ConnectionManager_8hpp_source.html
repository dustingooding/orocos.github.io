146  void select_reader_channel(Pred pred, bool copy_old_data) {
147  RTT::os::MutexLock lock(connection_lock);
148  std::pair<bool, ChannelDescriptor> new_channel =
149  find_if(pred, copy_old_data);
150  if (new_channel.first)
151  {
152  // We don't clear the current channel (to get it to NoData state), because there is a race
153  // between find_if and this line. We have to accept (in other parts of the code) that eventually,
154  // all channels return 'OldData'.
155  cur_channel = new_channel.second;
156  }
157  }

159  template<typename Pred>
160  std::pair<bool, ChannelDescriptor> find_if(Pred pred, bool copy_old_data) {
161  // We only copy OldData in the initial read of the current channel.
162  // if it has no new data, the search over the other channels starts,
163  // but no old data is needed.
164  ChannelDescriptor channel = cur_channel;
165  if ( channel.get<1>() )
166  if ( pred( copy_old_data, channel ) )
167  return std::make_pair(true, channel);
168 
169  std::list<ChannelDescriptor>::iterator result;
170  for (result = connections.begin(); result != connections.end(); ++result) {
171  if (result->get<1>() == cur_channel.get<1>()) continue;
172  if ( pred(false, *result) == true)
173  return std::make_pair(true, *result);
174  }
175  return std::make_pair(false, ChannelDescriptor());
176  }
