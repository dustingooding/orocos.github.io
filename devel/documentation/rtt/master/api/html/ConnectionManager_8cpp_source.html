138  bool ConnectionManager::removeConnection(ConnID* conn_id)
139  {
140  ChannelDescriptor descriptor;
141  { RTT::os::MutexLock lock(connection_lock);
142  std::list<ChannelDescriptor>::iterator conn_it =
143  std::find_if(connections.begin(), connections.end(), boost::bind(&ConnectionManager::findMatchingPort, this, conn_id, _1));
144  if (conn_it == connections.end())
145  return false;
146  descriptor = *conn_it;
147  connections.erase(conn_it);
148  updateCurrentChannel( cur_channel.get<1>() == descriptor.get<1>() );
149  }
150 
151  // disconnect needs to know if we're from Out->In (forward) or from In->Out
152  bool is_forward = true;
153  if ( dynamic_cast<InputPortInterface*>(mport) )
154  is_forward = false; // disconnect on input port = backward.
155 
156  descriptor.get<1>()->disconnect(is_forward);
157  return true;
158  }
