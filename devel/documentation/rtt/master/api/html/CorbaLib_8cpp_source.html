151  bool CorbaLibPlugin::registerTransport(std::string name, TypeInfo* ti)
152  {
153  if ( name == "unknown_t") // register fallback also.
154  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaFallBackProtocol());
155  if ( name == "int" )
156  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<int>() );
157  if ( name == "double" )
158  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<double>() );
159  if ( name == "float" )
160  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<float>() );
161  if ( name == "uint" )
162  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<unsigned int>() );
163  if ( name == "char" )
164  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<char>() );
165  //if ( name == "long" )
166  // return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<long>() );
167 // if ( name == "PropertyBag" )
168 // return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<PropertyBag>() );
169  if ( name == "bool" )
170  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<bool>() );
171 #ifndef RTT_NO_STD_TYPES
172  if ( name == "string" )
173  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<std::string>() );
174  if ( name == "array" )
175  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol< std::vector<double> >() );
176 #endif
177 #ifdef OS_RT_MALLOC
178  if ( name == "rt_string")
179  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<rt_string>() );
180 #endif
181  if ( name == "void" )
182  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaFallBackProtocol(false)); // warn=false
183  if ( name == "ConnPolicy")
184  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<ConnPolicy>() );
185  if ( name == "TaskContext")
186  return ti->addProtocol(ORO_CORBA_PROTOCOL_ID, new CorbaTemplateProtocol<TaskContext*>() );
187 
188  return false;
189  }
