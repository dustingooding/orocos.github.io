87  DataSourceBase::shared_ptr TypeInfo::construct(const std::vector<DataSourceBase::shared_ptr>& args) const
88  {
89 
90  DataSourceBase::shared_ptr ds;
91  // build default constructible:
92  if ( args.empty() ) {
93  AttributeBase* ab = this->buildVariable("constructor");
94  ds = ab->getDataSource();
95  delete ab;
96  return ds;
97  }
98 
99  // return same type if equal:
100  if ( args.size() == 1 && args.front()->getTypeInfo() == this )
101  return args.front();
102 
103  Constructors::const_iterator i= constructors.begin();
104  while (i != constructors.end() ) {
105  ds = (*i)->build( args );
106  if ( ds )
107  return ds;
108  ++i;
109  }
110  // returns empty data source to indicate not constructible
111  return ds;

136  bool TypeInfo::addProtocol(int protocol_id, TypeTransporter* tt)
137  {
138  if (transporters.size() < static_cast<size_t>(protocol_id + 1))
139  transporters.resize(protocol_id + 1);

149  TypeTransporter* TypeInfo::getProtocol(int protocol_id) const
150  {
151  // if the protocol is unknown to this type, return the protocol of the 'unknown type'
152  // type, which is a fallback such that we won't have to return zero, but can
153  // gracefully fall-back.
154  // In order to not endlessly recurse, we check if we aren't the UnknownType !
155  if ( protocol_id + 1 > int(transporters.size()) || transporters[protocol_id] == 0) {
156  if ( DataSourceTypeInfo<UnknownType>::getTypeInfo() != this )
157  return DataSourceTypeInfo<UnknownType>::getTypeInfo()->getProtocol( protocol_id );
158  else {
159  log(Warning) << "The protocol with id "<<protocol_id<<" did not register a fall-back handler for unknown types!"<<endlog();
160  log(Warning) << " triggered by: "<< getTypeName() << " which does not have a transport."<<endlog();
161  return 0; // That transport did not register a fall-back !
162  }
163  }
164  return transporters[protocol_id];
165  }
