98  virtual bool composeType(base::DataSourceBase::shared_ptr source,
99  base::DataSourceBase::shared_ptr result) const
100  {
101  // First, try a plain update.
102  if (result->update(source.get()))
103  return true;
104  // try conversion from int to enum:
105  internal::DataSource<int>::shared_ptr ds =
106  internal::DataSource<int>::narrow( source.get() );
107  if (ds)
108  {
109  typename internal::AssignableDataSource<T>::shared_ptr menum =
110  internal::AssignableDataSource<T>::narrow( source.get() );
111  assert(menum);
112  menum->set( (T)ds->get() );
113  return true;
114  }
115  // try conversion from string to enum:
116  internal::DataSource<std::string>::shared_ptr dss =
117  internal::DataSource<std::string>::narrow( source.get() );
118  if (dss)
119  {
120  typename internal::AssignableDataSource<T>::shared_ptr menum =
121  internal::AssignableDataSource<T>::narrow( result.get() );
122  assert(menum);
123  for( typename MapType::const_iterator it = to_string.begin(); it != to_string.end(); ++it)
124  if ( it->second == dss->get() ) {
125  menum->set( it->first );
126  return true;
127  }
128  }
129  // no conversion possible.
130  return false;
131  }

140  virtual base::DataSourceBase::shared_ptr decomposeType(base::DataSourceBase::shared_ptr source) const
141  {
142  // Convert enum to string
143  typename internal::DataSource<T>::shared_ptr ds = internal::DataSource<T>::narrow( source.get() );
144  if ( ds )
145  {
146  // if not available, just convert to int.
147  if ( to_string.count( ds->get() ) == 0 ) {
148  //log(Warning) << "No enum-to-string mapping defined for enum " << this->getTypeName() <<". Converting to int."<<endlog();
149  return new internal::ValueDataSource<int>( ds->get() );
150  }
151  internal::ValueDataSource<std::string>::shared_ptr vds = new internal::ValueDataSource<std::string>( to_string.find(ds->get())->second );
152  return vds;
153  }
154  // convert string to enum
