83  bool Service::addService( Service::shared_ptr obj ) {
84  if ( services.find( obj->getName() ) != services.end() ) {
85  log(Error) << "Could not add Service " << obj->getName() <<": name already in use." <<endlog();
86  return false;
87  }
88  // we pass and store a shared ptr in setParent, so we hack it like this:
89  shared_ptr me;
90  try {
91  me = shared_from_this();
92  } catch ( boost::bad_weak_ptr& bw ) {
93  me.reset(this); // take ownership
94  }
95  obj->setParent( me );
96 
97  // only set the owner if one is present.
98  if ( mowner ) {
99  obj->setOwner( mowner );
100  }
101  services[obj->getName()] = obj;
102  return true;
103  }

187  bool Service::addLocalOperation( OperationBase& op )
188  {
189  Logger::In in("Service::addLocalOperation");
190  if ( op.getName().empty() ) {
191  log(Error) << "Failed to add Operation: '"<< op.getName() <<"' has no name." <<endlog();
192  return false;
193  }
194  // don't check ready() since the op may not have an owner yet:
195  if ( !op.getImplementation() ) {
196  log(Error) << "Failed to add Operation: '"<< op.getName() <<"' is not ready: not bound to a function." <<endlog();
197  return false;
198  }
199  if ( simpleoperations.count( op.getName() ) ) {
200  log(Warning) << "While adding Operation: '"<< op.getName() <<"': replacing previously added operation." <<endlog();
201  this->removeOperation(op.getName());
202  }
203  simpleoperations[op.getName()] = &op;
204  // finally set the (new) owner:
205  if (mowner) {
206  // also updates the Executor:
207  op.setOwner(mowner->engine());
208  }
209  return true;
210  }
