80  {
81  struct timeval now;

84  tp->tv_sec = 0;

180  static inline int rtos_sem_wait_until(rt_sem_t* m, NANO_TIME abs_time )
181  {
182  TIME_SPEC timevl, delayvl;
183  TIME_SPEC arg_time = ticks2timespec( abs_time );
184  clock_gettime(CLOCK_REALTIME, &timevl);
185 
187 
188  // calculate delay from abs_time
189  delayvl.tv_sec = arg_time.tv_sec - timevl.tv_sec;
190  delayvl.tv_nsec = arg_time.tv_nsec - timevl.tv_nsec;
191  // tv_nsec is signed long in 10.6 (see sys/_structs.h)
192  if ( delayvl.tv_nsec >= 1000000000) {
193  ++delayvl.tv_sec;
194  delayvl.tv_nsec -= 1000000000;
195  }
196  if ( delayvl.tv_nsec < 0) {
197  --delayvl.tv_sec;
198  delayvl.tv_nsec += 1000000000;
199  }
200 
201  assert( 0 <= delayvl.tv_sec);
202  assert( 0 <= delayvl.tv_nsec);
203  assert( delayvl.tv_nsec < 1000000000 );
204 
205  mach_timespec_t mach_delayvl = { delayvl.tv_sec, delayvl.tv_nsec };
206  int rc = semaphore_timedwait( *m, mach_delayvl);
207  // map to return values from gnulinux, and expected by the calling layer
208  return (KERN_OPERATION_TIMED_OUT == rc ? -1 : 0);
209  }
