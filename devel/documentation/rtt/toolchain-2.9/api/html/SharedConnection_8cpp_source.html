117 void SharedConnectionRepository::remove(SharedConnectionBase* connection)
118 {
119  RTT::os::MutexLock lock(mutex);
120  for(Map::iterator it = map.begin(); it != map.end(); )
121  {
122  if (it->second == connection) {
123  Map::iterator copy = it++;
124  map.erase(copy);

126  it++;
