152  ,OROCOS_SIGNATURE_ARGS

154  ) );

158  if (this->emitting)
159  return NA<R>::na(); // avoid uglyness : Handlers calling emit.

162  const_iterator end = mconnections.end();

164  connection_impl* ci = static_cast<connection_impl*>( it->get() );
