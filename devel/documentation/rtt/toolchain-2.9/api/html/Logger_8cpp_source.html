180  void logit(std::ostream& (*pf)(std::ostream&))
181  {
182  // only on Logger::nl or Logger::endl, a time+log-line is written.
183  os::MutexLock lock( inpguard );
184  std:: string res = showTime() +" " + showLevel(inloglevel) + showModule() + " ";
185 
186  // do not log if not wanted.
187  if ( maylogStdOut() ) {
188 #ifndef OROSEM_PRINTF_LOGGING
189  *stdoutput << res << logline.str() << pf;
190 #else
191  printf("%s%s\n", res.c_str(), logline.str().c_str() );
192 #endif
193  logline.str(""); // clear stringstream.
194  }
195 
196  if ( maylogFile() ) {
197 #ifdef OROSEM_FILE_LOGGING
198 #if defined(OROSEM_LOG4CPP_LOGGING)
199  category.log(level2Priority(inloglevel), fileline.str());
200 #elif !defined(OROSEM_PRINTF_LOGGING)
201  logfile << res << fileline.str() << pf;
202 #else
203  fprintf( logfile, "%s%s\n", res.c_str(), fileline.str().c_str() );
204 #endif
205 #ifdef OROSEM_REMOTE_LOGGING
206  // detect buffer 'overflow'
207  if ( messagecnt >= ORONUM_LOGGING_BUFSIZE ) {
208  std::string dummy;
209  remotestream >> dummy; // FIFO principle: read 1 line
210  --messagecnt;
211  }
212  remotestream << res << fileline.str() << pf;
213  ++messagecnt;
214 #endif
215 #if defined(OROSEM_FILE_LOGGING) || defined(OROSEM_REMOTE_LOGGING)
216  fileline.str("");
217 #endif
218 #endif
219  }
220  }

445  void Logger::startup() {
446  if (d->started)
447  return;
448 #ifndef OROBLD_DISABLE_LOGGING
449  std::string xtramsg = "No ORO_LOGLEVEL environment variable set.";
450  *this << Logger::Info; // default log to Info
451 
452  int wantedlevel=4; // default log level is 4.
453 
454  if ( getenv( "ORO_LOGLEVEL" ) != 0 ) {
455  std::stringstream conv;
456  conv.str( std::string( getenv( "ORO_LOGLEVEL" ) ) );
457  conv >> wantedlevel;
458  if ( conv.fail() ) {
459  xtramsg = std::string( "Failed to extract loglevel from environment variable ORO_LOGLEVEL.")
460  + " It contained the string '"+conv.str()+"', while it should contain an integer value.";
461  *this<<Logger::Error;
462  }
463  else {
464  d->outloglevel = d->intToLogLevel(wantedlevel);
465  xtramsg = "Successfully extracted environment variable ORO_LOGLEVEL";
466  }
467  }
468 
469  // Completely disable logging on negative values.
470  if ( wantedlevel < 0 )
471  return;
472  d->started = true;
473 
474  d->timestamp = TimeService::Instance()->getTicks();
475  *this<<xtramsg<<Logger::nl;
476  *this<< " OROCOS version '" ORO_xstr(RTT_VERSION) "'";
477 #ifdef __GNUC__
478  *this << " compiled with GCC " ORO_xstr(__GNUC__) "." ORO_xstr(__GNUC_MINOR__) "." ORO_xstr(__GNUC_PATCHLEVEL__) ".";
479 #endif
480 #ifdef OROPKG_OS_LXRT
481  *this<<" Running in LXRT/RTAI."<< Logger::nl;
482 #endif
483 #ifdef OROPKG_OS_GNULINUX
484  *this<<" Running in GNU/Linux."<< Logger::nl;
485 #endif
486 #ifdef OROPKG_OS_XENOMAI
487  *this<<" Running in Xenomai."<< Logger::nl;
488 #endif
489  *this<<"Orocos Logging Activated at level : " << d->showLevel( d->outloglevel ) << " ( "<<int(d->outloglevel)<<" ) "<< Logger::nl;
490  *this<<"Reference System Time is : " << d->timestamp << " ticks ( "<< Seconds(TimeService::ticks2nsecs(d->timestamp))/NSECS_IN_SECS <<" seconds )." << Logger::nl;
491  *this<<"Logging is relative to this time." <<Logger::endl;
492 #endif
493  }
