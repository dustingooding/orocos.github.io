152  static ds_type sources(std::vector<base::DataSourceBase::shared_ptr>::const_iterator front, int argnbr, std::string const& tname )
153  {
154  typedef typename ds_type::element_type element_type;
155 
156  ds_type a =
157  boost::dynamic_pointer_cast< element_type >( DataSourceTypeInfo<ds_arg_type>::getTypeInfo()->convert(*front) );
158  if ( ! a ) {
159  //cout << typeid(DataSource<ds_arg_type>).name() << endl;

162  }
163  return a;
164  }
