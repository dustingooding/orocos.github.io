156  template<typename T>
157  static base::ChannelElement<T>* buildDataStorage(ConnPolicy const& policy, const T& initial_value = T())
158  {
159  if (policy.type == ConnPolicy::DATA)
160  {
161  typename base::DataObjectInterface<T>::shared_ptr data_object;
162  switch (policy.lock_policy)
163  {
164 #ifndef OROBLD_OS_NO_ASM
165  case ConnPolicy::LOCK_FREE:
166  // DataObjectLockFree does not support multiple writers. We have to enforce lock policy LOCKED in this case.
167  if (policy.buffer_policy == PerInputPort || policy.buffer_policy == Shared) {
168  RTT::log(Error) << "Lock-free data objects do not allow multiple writers at this moment and therefore cannot be used in connection with the PerInputPort or Shared buffer policies." << endlog();
169  return NULL;
170  }
171  data_object.reset( new base::DataObjectLockFree<T>(initial_value, policy) );
172  break;
173 #else
174  case ConnPolicy::LOCK_FREE:
175  RTT::log(Warning) << "lock free connection policy is unavailable on this system, defaulting to LOCKED" << RTT::endlog();
176 #endif
177  case ConnPolicy::LOCKED:
178  data_object.reset( new base::DataObjectLocked<T>(initial_value) );
179  break;
180  case ConnPolicy::UNSYNC:
181  data_object.reset( new base::DataObjectUnSync<T>(initial_value) );
182  break;
183  }
184  return new ChannelDataElement<T>(data_object, policy);
185  }
186  else if (policy.type == ConnPolicy::BUFFER || policy.type == ConnPolicy::CIRCULAR_BUFFER)
187  {
188  typename base::BufferInterface<T>::shared_ptr buffer_object;
189  switch (policy.lock_policy)
190  {
191 #ifndef OROBLD_OS_NO_ASM
192  case ConnPolicy::LOCK_FREE:
193  buffer_object.reset(new base::BufferLockFree<T>(policy.size, initial_value, policy));
194  break;
195 #else
196  case ConnPolicy::LOCK_FREE:
197  RTT::log(Warning) << "lock free connection policy is unavailable on this system, defaulting to LOCKED" << RTT::endlog();
198 #endif
199  case ConnPolicy::LOCKED:
200  buffer_object.reset(new base::BufferLocked<T>(policy.size, initial_value, policy));
201  break;
202  case ConnPolicy::UNSYNC:
203  buffer_object.reset(new base::BufferUnSync<T>(policy.size, initial_value, policy));
204  break;
205  }
206  return new ChannelBufferElement<T>(buffer_object, policy);
207  }
208  return NULL;

224  static base::ChannelElementBase::shared_ptr buildChannelInput(OutputPort<T>& port, ConnPolicy const& policy, bool force_unbuffered = false)
225  {
226  typename internal::ConnInputEndpoint<T>::shared_ptr endpoint = port.getEndpoint();
227  typename base::ChannelElement<T>::shared_ptr buffer = port.getSharedBuffer();
228 
229  // Set buffer policy of the port
230  if (!(endpoint->setBufferPolicy(policy.buffer_policy))) {
231  log(Error) << "You mixed incompatible buffer policies for output port " << port.getName() << ": "
232  << "The new connection requests a " << policy.buffer_policy << " policy, "
233  << "but the port already has a " << endpoint->getBufferPolicy() << " policy." << endlog();
234  return typename internal::ConnOutputEndpoint<T>::shared_ptr();
235  }
236 
237  // Note: PerInputPort implies PUSH and PerOutputPort implies PULL
238  bool pull = policy.pull;
239  if (policy.buffer_policy == PerInputPort) pull = ConnPolicy::PUSH;
240  if (policy.buffer_policy == PerOutputPort) pull = ConnPolicy::PULL;
241 
242  if (pull == ConnPolicy::PULL && !force_unbuffered) {
243  if (!buffer) {
244  buffer = buildDataStorage<T>(policy, port.getLastWrittenValue());
245  if (!buffer) return typename internal::ConnOutputEndpoint<T>::shared_ptr();
246 
247  if (policy.buffer_policy == PerOutputPort) {
248  // For PerOutputPort connections, the buffer is installed BEFORE the input endpoint!
249  if (endpoint->connected()) {
250  log(Error) << "You tried to create a shared output buffer connection for output port " << port.getName() << ", "
251  << "but the port already has at least one incompatible outgoing connection." << endlog();
252  return typename internal::ConnOutputEndpoint<T>::shared_ptr();
253  }
254  return buffer->connectTo(endpoint) ? endpoint : typename internal::ConnInputEndpoint<T>::shared_ptr();
255  } else {
256  return endpoint->connectTo(buffer, policy.mandatory) ? buffer : typename internal::ConnInputEndpoint<T>::shared_ptr();
257  }
258 
259  } else if (policy.buffer_policy == PerOutputPort) {
260  // check that the existing buffer type is compatible to the new ConnPolicy
261  assert(buffer->getConnPolicy());
262  ConnPolicy buffer_policy = *(buffer->getConnPolicy());
263 
264  if (
265  (buffer_policy.type != policy.type) ||
266  (buffer_policy.size != policy.size) ||
267  (buffer_policy.lock_policy != policy.lock_policy)
268  )
269  {
270  log(Error) << "You mixed incompatible connection policies for the shared output buffer of port " << port.getName() << ": "
271  << "The new connection requests a " << policy << " connection, "
272  << "but the port already has a " << buffer_policy << " buffer." << endlog();
273  return typename internal::ConnOutputEndpoint<T>::shared_ptr();
274  }
275 
276  return endpoint;
277  }
278  }
279 
280  if (buffer) {
281  // The new connection requires an unbuffered channel input or a per-connection buffer, but this port already has a shared output buffer!
282  assert(buffer->getConnPolicy());
283  ConnPolicy buffer_policy = *(buffer->getConnPolicy());
284 
285  log(Error) << "You mixed incompatible connection policies for output port " << port.getName() << ": "
286  << "The new connection requests a " << policy << " connection, "
287  << "but the port already has a " << buffer_policy << " buffer." << endlog();
288  return typename internal::ConnOutputEndpoint<T>::shared_ptr();
289  }
290 
291  return endpoint;

412  return SharedConnectionBase::shared_ptr();
413  }
414

416  if (input_port && !input_port->isLocal()) {

470  template<typename T>
471  static bool createConnection(OutputPort<T>& output_port, base::InputPortInterface& input_port, ConnPolicy const& policy)
472  {
473  if ( !output_port.isLocal() ) {
474  log(Error) << "Need a local OutputPort to create connections." <<endlog();
475  return false;
476  }
477 
478  if (output_port.connectedTo(&input_port)) {
479  log(Info) << "OutputPort " << output_port.getName() << " is already connected to " << input_port.getName() << ", ignoring new connection." << endlog();
480  return true;
481  }
482 
483  InputPort<T>* input_p = dynamic_cast<InputPort<T>*>(&input_port);
484 
485  // Shared push connection? => forward to createAndCheckSharedConnection()
486  if (policy.buffer_policy == Shared) {
487  return createAndCheckSharedConnection(&output_port, &input_port, buildSharedConnection<T>(&output_port, &input_port, policy), policy);
488  }
489 
490  // This is the input and output channel element of the output half
491  base::ChannelElementBase::shared_ptr output_half;
492  if (input_port.isLocal() && policy.transport == 0)
493  {
494  // Local connection
495  if (!input_p)
496  {
497  log(Error) << "Port " << input_port.getName() << " is not compatible with " << output_port.getName() << endlog();
498  return false;
499  }
500 
501  // local ports, create buffer here.
502  output_half = buildChannelOutput<T>(*input_p, policy, output_port.getLastWrittenValue());
503  }
504  else
505  {
506  // if the input is not local, this is a pure remote connection,
507  // if the input *is* local, the user requested to use a different transport
508  // than plain memory, rare case, but we accept it. The unit tests use this for example
509  // to test the OOB transports.
510  if ( !input_port.isLocal() ) {
511  output_half = buildRemoteChannelOutput( output_port, input_port, policy);
512  } else if (input_p) {
513  return createOutOfBandConnection<T>( output_port, *input_p, policy);
514  } else {
515  log(Error) << "Port " << input_port.getName() << " is not compatible with " << output_port.getName() << endlog();
516  return false;
517  }
518  }
519 
520  if (!output_half)
521  return false;
522 
523  // Since output is local, buildChannelInput is local as well.
524  // This this the input channel element of the whole connection
525  base::ChannelElementBase::shared_ptr channel_input;
526  channel_input = buildChannelInput<T>(output_port, policy);
527 
528  if (!channel_input) {
529  output_half->disconnect(true);
530  return false;
531  }
532 
533  // NOTE: channel_input and output_half are not yet connected!
534  return createAndCheckConnection(output_port, input_port, channel_input, output_half, policy);
535  }

588  template<class T>
589  static bool createOutOfBandConnection(OutputPort<T>& output_port, InputPort<T>& input_port, ConnPolicy const& policy) {
590  // constructs an out-of-band channel:
591  // output_port -> channel_input -> stream_input -> (out-of-band transport of data) -> stream_output -> channel_output -> input_port
592  // |-- (direct connection for coordination) --^
593  //
594 
595  // Stream channel inputs are always unbuffered (push). It's the transport that has to add a buffer element if required.
596  RTT::base::ChannelElementBase::shared_ptr channel_input = buildChannelInput( output_port, policy, /* force_unbuffered = */ true );
597  if (!channel_input) return false;
598 
599  RTT::base::ChannelElementBase::shared_ptr stream_input = createAndCheckStream(output_port, policy, channel_input, new StreamConnID(policy.name_id));
600  if (!stream_input) return false;
601 
602  RTT::base::ChannelElementBase::shared_ptr channel_output = ConnFactory::buildChannelOutput<T>(input_port, policy, output_port.getLastWrittenValue());
603  if (!channel_output) return false;
604 
605  RTT::base::ChannelElementBase::shared_ptr stream_output = createAndCheckStream(input_port, policy, channel_output, new StreamConnID(policy.name_id));
606  if (!stream_output) return false;
607 
608  return stream_input->getOutputEndPoint()->connectTo(stream_output->getInputEndPoint(), policy.mandatory);
609  }
