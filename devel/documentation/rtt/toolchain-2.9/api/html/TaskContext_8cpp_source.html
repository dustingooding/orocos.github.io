118  TaskContext::~TaskContext()
119  {
120  if (our_act)
121  our_act->stop();
122  // We don't call stop() or cleanup() here since this is
123  // the responsibility of the subclass. Calling these functions
124  // here would only lead to calling invalid virtual functions.
125  // [Rule no 1: Don't call virtual functions in a destructor.]
126  // [Rule no 2: Don't call virtual functions in a constructor.]
127 
128  // these need to be freed before we cleanup the EE:
129  localservs.clear();
130  tcservice.reset();
131  tcrequests.reset();
132 
133  // remove from all users.
134  while( !musers.empty() ) {
135  musers.front()->removePeer(this);
136  }
137  // since we are destroyed, be sure that the peer no longer
138  // has a 'user' pointer to us.
139  while ( !_task_map.empty() ) {
140  _task_map.begin()->second->removeUser(this);
141  _task_map.erase( _task_map.begin() );
142  }
143  // Do not call this->disconnect() !!!
144  // Ports are probably already destructed by user code.
145  }

147  bool TaskContext::connectPorts( TaskContext* peer )
148  {
149  bool failure = false;
150  const std::string& location = this->getName();
151  Logger::In in( location.c_str() );
152 
153  DataFlowInterface::Ports myports = this->ports()->getPorts();
154  for (DataFlowInterface::Ports::iterator it = myports.begin();
155  it != myports.end();
156  ++it) {
157 
158  // Then try to get the peer port's connection
159  PortInterface* peerport = peer->ports()->getPort( (*it)->getName() );
160  if ( !peerport ) {
161  log(Debug)<< "Peer Task "<<peer->getName() <<" has no Port " << (*it)->getName() << endlog();
162  continue;
163  }
164 
165  // Skip if they have the same type
166  if((dynamic_cast<OutputPortInterface*>(*it) && dynamic_cast<OutputPortInterface*>(peerport)) ||
167  (dynamic_cast<InputPortInterface*>(*it) && dynamic_cast<InputPortInterface*>(peerport)))
168  {
169  log(Debug)<< (*it)->getName() << " and " << peerport->getName() << " have the same type" << endlog();
170  continue;
171  }
172 
173  // Try to find a way to connect them
174  if ( !(*it)->connectTo( peerport ) ) {
175  log(Debug)<< "Data flow incompatible between ports "
176  << getName() << "." << (*it)->getName() << " and "
177  << peer->getName() << "." << (*it)->getName() << endlog();
178  failure = true;
179  }
180  }
181  return !failure;
182  }

184  bool TaskContext::connectServices( TaskContext* peer )
185  {
186  bool success = true;
187  const std::string& location = this->getName();
188  Logger::In in( location.c_str() );
189 
190  vector<string> myreqs = this->requires()->getRequesterNames();
191  vector<string> peerreqs = peer->requires()->getRequesterNames();
192 
193  this->requires()->connectTo( peer->provides() );
194  for (vector<string>::iterator it = myreqs.begin();
195  it != myreqs.end();
196  ++it) {
197  ServiceRequester::shared_ptr sr = this->requires(*it);
198  if ( !sr->ready() ) {
199  if (peer->provides()->hasService( *it ))
200  success = sr->connectTo( peer->provides(*it) ) && success;
201  else {
202  log(Debug)<< "Peer Task "<<peer->getName() <<" provides no Service " << *it << endlog();
203  }
204  }
205  }
206 
207  peer->requires()->connectTo( this->provides() );
208  for (vector<string>::iterator it = peerreqs.begin();
209  it != peerreqs.end();
210  ++it) {
211  ServiceRequester::shared_ptr sr = peer->requires(*it);
212  if ( !sr->ready() ) {
213  if (this->provides()->hasService(*it))
214  success = sr->connectTo( this->provides(*it) ) && success;
215  else
216  log(Debug)<< "This Task provides no Service " << *it << " for peer Task "<<peer->getName() <<"."<< endlog();
217  }
218  }
219  return success;
220  }

285  void TaskContext::disconnect() {
286  Logger::In in( this->getName().c_str() );
287  // disconnect all our ports
288  DataFlowInterface::Ports myports = this->ports()->getPorts();
289  for (DataFlowInterface::Ports::iterator it = myports.begin();
290  it != myports.end();
291  ++it) {
292  (*it)->disconnect();
293  }
294 
295  // remove from all users.
296  while( !musers.empty() ) {
297  musers.front()->removePeer(this);
298  }
299 
300  while ( !_task_map.empty() ) {
301  _task_map.begin()->second->removeUser(this);
302  _task_map.erase( _task_map.begin() );
303  }
304  }
