85  virtual FlowStatus read(reference_t sample, bool copy_old_data)
86  {
87  value_t *new_sample_p;
88  if ( (new_sample_p = buffer->PopWithoutRelease()) ) {
89  if(last_sample_p)
90  buffer->Release(last_sample_p);
91 
92  sample = *new_sample_p;
93 
94  // In the PerOutputPort or Shared buffer policy case this buffer element may be read by multiple readers.
95  // ==> We cannot store the last_sample and release immediately.
96  // ==> WriteShared buffer connections will never return OldData.
97  if (policy.buffer_policy != PerOutputPort && policy.buffer_policy != Shared)
98  last_sample_p = new_sample_p;
99  else
100  buffer->Release(new_sample_p);
101 
102  return NewData;
103  }
104  if (last_sample_p) {
105  if(copy_old_data)
106  sample = *(last_sample_p);
107  return OldData;
108  }
109  return NoData;
110  }
