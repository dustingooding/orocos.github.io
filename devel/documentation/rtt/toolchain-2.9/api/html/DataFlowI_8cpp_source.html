257 bool CDataFlowInterface_i::removeConnection(
258  const char* local_port,
259  CDataFlowInterface_ptr remote_interface, const char* remote_port) ACE_THROW_SPEC ((
260  CORBA::SystemException
261  ,::RTT::corba::CNoSuchPortException
262  ))
263 {
264  PortInterface* port = mdf->getPort(local_port);
265  // CORBA does not support disconnecting from the input port
266  if (port == 0) {
267  log(Error) << "disconnectPort: No such port: "<< local_port <<endlog();
268  throw corba::CNoSuchPortException();
269  }
270  if (dynamic_cast<OutputPortInterface*>(port) == 0) {
271  log(Error) << "disconnectPort: "<< local_port << " is an input port" << endlog();
272  throw corba::CNoSuchPortException();
273  }
274 
275  RTT::DataFlowInterface* local_interface = CDataFlowInterface_i::getLocalInterface(remote_interface);
276  if (local_interface)
277  {
278  PortInterface* other_port = local_interface->getPort(remote_port);
279  if (!other_port)
280  return false;
281 
282  // Try to disconnect the local port. However, one might have forced
283  // having a CORBA connection between local ports, so if it fails go on
284  // with normal CORBA disconnection
285  if (port->disconnect(other_port))
286  return true;
287 
288  }
289 
290  CORBA_CHECK_THREAD();
291  RemoteConnID rcid(remote_interface, remote_port);
292  return port->removeConnection( &rcid );
293 }

332 CChannelElement_ptr CDataFlowInterface_i::buildChannelOutput(
333  const char* port_name, CConnPolicy & corba_policy) ACE_THROW_SPEC ((
334  CORBA::SystemException
335  ,::RTT::corba::CNoCorbaTransport
336  ,::RTT::corba::CNoSuchPortException
337  ,::RTT::corba::CInvalidArgument
338  ))
339 {
340  Logger::In in("CDataFlowInterface_i::buildChannelOutput");
341  InputPortInterface* port = dynamic_cast<InputPortInterface*>(mdf->getPort(port_name));
342  if (port == 0)
343  throw CNoSuchPortException();
344 
345  TypeInfo const* type_info = port->getTypeInfo();
346  if (!type_info)
347  throw CNoCorbaTransport();
348 
349  CorbaTypeTransporter* transporter =
350  dynamic_cast<CorbaTypeTransporter*>(type_info->getProtocol(ORO_CORBA_PROTOCOL_ID));
351  if (!transporter)
352  throw CNoCorbaTransport();
353 
354  CORBA_CHECK_THREAD();
355  // Convert to RTT policy.
356  ConnPolicy policy2 = toRTT(corba_policy);
357 
358  // For shared push connections, also build or check the local shared connection instance here
359  ChannelElementBase::shared_ptr end;
360  if (policy2.buffer_policy == Shared) {
361  internal::SharedConnectionBase::shared_ptr shared_connection = type_info->buildSharedConnection(0, port, policy2);
362  if (!shared_connection) {
363  throw CInvalidArgument();
364  }
365 
366  // If no user supplied name, pass on the new name.
367  if ( strlen( corba_policy.name_id.in()) == 0 )
368  corba_policy.name_id = CORBA::string_dup( shared_connection->getName().c_str() );
369 
370  // Connect the input port if it is not already connected
371  if (!port->createConnection(shared_connection, policy2)) {
372  return RTT::corba::CChannelElement::_nil();
373  }
374 
375  end = shared_connection;
376 
377  // ...otherwise, create default channel output.
378  } else {
379  end = type_info->buildChannelOutput(*port, policy2);
380  if (!end) throw CInvalidArgument();
381  }
382 
383  CRemoteChannelElement_i* this_element =
384  transporter->createChannelElement_i(mdf, mpoa, policy2);
385  this_element->setCDataFlowInterface(this);
386 
387  /*
388  * This part is for out-of band (needs to be factored out).
389  */
390  if ( corba_policy.transport !=0 && corba_policy.transport != ORO_CORBA_PROTOCOL_ID) {
391  // prepare out-of-band transport for this port.
392  // if user supplied name, use that one.
393  if ( type_info->getProtocol(corba_policy.transport) == 0 ) {
394  log(Error) << "Could not create out-of-band transport for port "<< port_name << " with transport id " << corba_policy.transport <<endlog();
395  log(Error) << "No such transport registered. Check your corba_policy.transport settings or add the transport for type "<< type_info->getTypeName() <<endlog();
396  return RTT::corba::CChannelElement::_nil();
397  }
398  RTT::base::ChannelElementBase::shared_ptr ceb = type_info->getProtocol(corba_policy.transport)->createStream(port, policy2, /* is_sender = */ false);
399  // if no user supplied name, pass on the new name.
400  if ( strlen( corba_policy.name_id.in()) == 0 )
401  corba_policy.name_id = CORBA::string_dup( policy2.name_id.c_str() );
402 
403  if (ceb) {
404  // override, insert oob element between corba and endpoint and add a buffer between oob and endpoint.
405  dynamic_cast<ChannelElementBase*>(this_element)->connectTo(ceb, policy2.mandatory);
406  ceb->getOutputEndPoint()->connectTo(end, policy2.mandatory);
407  log(Info) <<"Receiving data for port "<< policy2.name_id << " from out-of-band protocol "<< corba_policy.transport <<endlog();
408  } else {
409  log(Error) << "The type transporter for type "<<type_info->getTypeName()<< " failed to create an out-of-band endpoint for port " << port_name<<endlog();
410  return RTT::corba::CChannelElement::_nil();
411  }
412  //
413  } else {
414  // No OOB.
415  dynamic_cast<ChannelElementBase*>(this_element)->connectTo(end, policy2.mandatory);
416  }
417 
418  this_element->_remove_ref();
419 
420  // store our mapping of corba channel elements to C++ channel elements. We need this for channelReady() and removing a channel again.
421  { RTT::os::MutexLock lock(channel_list_mtx);
422  channel_list.push_back( ChannelList::value_type(this_element->_this(), end) );
423  }
424 
425  CRemoteChannelElement_var proxy = this_element->_this();
426  return proxy._retn();
427 }

432 CChannelElement_ptr CDataFlowInterface_i::buildChannelInput(
433  const char* port_name, CConnPolicy & corba_policy) ACE_THROW_SPEC ((
434  CORBA::SystemException
435  ,::RTT::corba::CNoCorbaTransport
436  ,::RTT::corba::CNoSuchPortException
437  ,::RTT::corba::CInvalidArgument
438  ))
439 {
440  Logger::In in("CDataFlowInterface_i::buildChannelInput");
441  // First check validity of user input...
442  OutputPortInterface* port = dynamic_cast<OutputPortInterface*>(mdf->getPort(port_name));
443  if (port == 0)
444  throw CNoSuchPortException();
445 
446  TypeInfo const* type_info = port->getTypeInfo();
447  if (!type_info)
448  throw CNoCorbaTransport();
449 
450  CorbaTypeTransporter* transporter =
451  dynamic_cast<CorbaTypeTransporter*>(type_info->getProtocol(ORO_CORBA_PROTOCOL_ID));
452  if (!transporter)
453  throw CNoCorbaTransport();
454 
455  CORBA_CHECK_THREAD();
456  // Convert to RTT policy.
457  ConnPolicy policy2 = toRTT(corba_policy);
458 
459  // Now create the output-side channel elements.
460  ChannelElementBase::shared_ptr start = type_info->buildChannelInput(*port, policy2);
461  if (!start) {
462  throw CInvalidArgument();
463  }
464 
465  // The channel element that exposes our channel in CORBA
466  CRemoteChannelElement_i* this_element;
467  PortableServer::ServantBase_var servant = this_element = transporter->createChannelElement_i(mdf, mpoa, policy2);
468  this_element->setCDataFlowInterface(this);
469  assert( dynamic_cast<ChannelElementBase*>(this_element) );
470 
471  /*
472  * This part if for out-of band. (needs to be factored out).
473  */
474  if ( corba_policy.transport !=0 && corba_policy.transport != ORO_CORBA_PROTOCOL_ID) {
475  // prepare out-of-band transport for this port.
476  // if user supplied name, use that one.
477  if ( type_info->getProtocol(corba_policy.transport) == 0 ) {
478  log(Error) << "Could not create out-of-band transport for port "<< port_name << " with transport id " << corba_policy.transport <<endlog();
479  log(Error) << "No such transport registered. Check your corba_policy.transport settings or add the transport for type "<< type_info->getTypeName() <<endlog();
480  throw CNoCorbaTransport();
481  }
482  RTT::base::ChannelElementBase::shared_ptr ceb = type_info->getProtocol(corba_policy.transport)->createStream(port, policy2, /* is_sender = */ true);
483  // if no user supplied name, pass on the new name.
484  if ( strlen( corba_policy.name_id.in()) == 0 )
485  corba_policy.name_id = CORBA::string_dup( policy2.name_id.c_str() );
486 
487  if (ceb) {
488  // OOB is added to end of chain.
489  start->connectTo( dynamic_cast<ChannelElementBase*>(this_element), policy2.mandatory );
490  dynamic_cast<ChannelElementBase*>(this_element)->connectTo( ceb );
491  log(Info) <<"Sending data from port "<< policy2.name_id << " to out-of-band protocol "<< corba_policy.transport <<endlog();
492  } else {
493  log(Error) << "The type transporter for type "<<type_info->getTypeName()<< " failed to create an out-of-band endpoint for port " << port_name<<endlog();
494  throw CNoCorbaTransport();
495  }
496 
497  } else {
498  // No OOB. Always add output buffer.
499  ChannelElementBase::shared_ptr buf = type_info->buildDataStorage(toRTT(corba_policy));
500  start->connectTo(buf, policy2.mandatory);
501  buf->connectTo( dynamic_cast<ChannelElementBase*>(this_element) );
502  }
503 
504 
505  // Attach to our output port:
506  port->addConnection( new SimpleConnID(), start->getInputEndPoint(), policy2 );
507 
508  // DO NOT DO THIS: _remove_ref() is tied to the refcount of the ChannelElementBase !
509  //this_element->_remove_ref();
510 
511  // Finally, store our mapping of corba channel elements to C++ channel elements. We need this for channelReady() and removing a channel again.
512  { RTT::os::MutexLock lock(channel_list_mtx);
513  channel_list.push_back( ChannelList::value_type(this_element->_this(), start->getInputEndPoint()));
514  }
515 
516  return this_element->_this();
517 }

555 ::CORBA::Boolean CDataFlowInterface_i::createConnection(
556  const char* writer_port, CDataFlowInterface_ptr reader_interface,
557  const char* reader_port, CConnPolicy & policy) ACE_THROW_SPEC ((
558  CORBA::SystemException
559  ,::RTT::corba::CNoSuchPortException
560  ))
561 {
562  Logger::In in("CDataFlowInterface_i::createConnection");
563  OutputPortInterface* writer = dynamic_cast<OutputPortInterface*>(mdf->getPort(writer_port));
564  if (writer == 0)
565  throw CNoSuchPortException();
566 
567  CORBA_CHECK_THREAD();
568  // Check if +reader_interface+ is local. If it is, use the non-CORBA
569  // connection.
570  RTT::DataFlowInterface* local_interface = CDataFlowInterface_i::getLocalInterface(reader_interface);
571  if (local_interface && policy.transport == 0)
572  {
573  InputPortInterface* reader =
574  dynamic_cast<InputPortInterface*>(local_interface->getPort(reader_port));
575  if (!reader)
576  {
577  log(Warning) << "CORBA: createConnection() target is not an input port" << endlog();
578  throw CNoSuchPortException();
579  return false;
580  }
581 
582  log(Debug) << "CORBA: createConnection() is creating a LOCAL connection between " <<
583  writer_port << " and " << reader_port << endlog();
584  return writer->createConnection(*reader, toRTT(policy));
585  }
586  else
587  log(Debug) << "CORBA: createConnection() is creating a REMOTE connection between " <<
588  writer_port << " and " << reader_port << endlog();
589 
590  try {
591  if (reader_interface->getPortType(reader_port) != corba::CInput) {
592  log(Error) << "Could not create connection: " << reader_port <<" is not an input port."<<endlog();
593  throw CNoSuchPortException();
594  return false;
595  }
596 
597  // Creates a proxy to the remote input port
598  RemoteInputPort port(writer->getTypeInfo(), reader_interface, reader_port, mpoa);
