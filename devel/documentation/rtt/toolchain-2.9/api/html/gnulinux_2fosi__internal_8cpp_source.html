106  INTERNAL_QUAL int rtos_task_create(RTOS_TASK* task,
107  int priority,
108  unsigned cpu_affinity,
109  const char * name,
110  int sched_type,
111  size_t stack_size,
112  void * (*start_routine)(void *),
113  ThreadInterface* obj)
114  {
115  int rv; // return value
116  task->wait_policy = ORO_WAIT_ABS;
117  rtos_task_check_priority( &sched_type, &priority );
118  // Save priority internally, since the pthread_attr* calls are broken !
119  // we will pick it up later in rtos_task_set_scheduler().
120  task->priority = priority;
121 
122  PosixCookie* xcookie = (PosixCookie*)malloc( sizeof(PosixCookie) );
123  xcookie->data = obj;
124  xcookie->wrapper = start_routine;
125 
126  // Set name
127  if ( strlen(name) == 0 )
128  name = "Thread";
129  task->name = strcpy( (char*)malloc( (strlen(name) + 1) * sizeof(char)), name);
130 
131  if ( (rv = pthread_attr_init(&(task->attr))) != 0 ){
132  return rv;
133  }
134  // Set scheduler type (_before_ assigning priorities!)
135  if ( (rv = pthread_attr_setschedpolicy(&(task->attr), sched_type)) != 0){
136  return rv;
137  }
138  // Set stack size
139  if (stack_size )
140  if ( (rv = pthread_attr_setstacksize(&(task->attr), stack_size)) != 0){
141  return rv;
142  }
143  pthread_attr_getschedpolicy(&(task->attr), &rv );
144  assert( rv == sched_type );
145  /* SCHED_OTHER tasks are always assigned static priority 0, see
146  man sched_setscheduler
147  */
148  struct sched_param sp;
149  if (sched_type != SCHED_OTHER){
150  sp.sched_priority=priority;
151  // Set priority
152  if ( (rv = pthread_attr_setschedparam(&(task->attr), &sp)) != 0 ){
153  return rv;
154  }
155  }
156  rv = pthread_create(&(task->thread), &(task->attr),
157  rtos_posix_thread_wrapper, xcookie);
158  if (rv != 0) {
159  log(Error) << "Failed to create thread " << task->name << ": "
160  << strerror(rv) << endlog();
161  return rv;
162  }
163 
164  // Set thread name to match task name, to help with debugging
165  {
166  // trim the name to fit 16 bytes restriction (including terminating
167  // \0 character) of pthread_setname_np
168  static const int MAX_THREAD_NAME_SIZE = 15;
169  const char *thread_name = task->name;
170  std::size_t thread_name_len = strlen(thread_name);
171  if (thread_name_len > MAX_THREAD_NAME_SIZE) {
172  thread_name += thread_name_len - MAX_THREAD_NAME_SIZE;
173  }
174  int result = pthread_setname_np(task->thread, thread_name);
175  if (result != 0) {
176  log(Error) << "Failed to set thread name for " << task->name << ": "
177  << strerror(result) << endlog();
178  }
179  }
180 
181  if ( cpu_affinity != (unsigned)~0 ) {
182  log(Debug) << "Setting CPU affinity to " << cpu_affinity << endlog();
183  int result = rtos_task_set_cpu_affinity(task, cpu_affinity);
184  if (result != 0) {
185  log(Error) << "Failed to set CPU affinity to " << cpu_affinity << " for " << task->name << ": "
186  << strerror(result) << endlog();
187  }
188  }
189 
190  return rv;
191  }

265  INTERNAL_QUAL int rtos_task_wait_period( RTOS_TASK* task )
266  {
267  if ( task->period == 0 )
268  return 0;
269 
270  // record this to detect overrun.
271  NANO_TIME now = rtos_get_time_monotonic_ns();
272  NANO_TIME wake= task->periodMark.tv_sec * 1000000000LL + task->periodMark.tv_nsec;
273 
274  // inspired by nanosleep man page for this construct:
275  while ( clock_nanosleep(CLOCK_MONOTONIC, TIMER_ABSTIME, &(task->periodMark), NULL) != 0 && errno == EINTR ) {
276  errno = 0;
277  }
278 
279  if (task->wait_policy == ORO_WAIT_ABS)
280  {
281  // in the case of overrun by more than 4 periods,
282  // skip all the updates before now, with the next update aligned to period
283  int maxDelayInPeriods = 4;
284  NANO_TIME period = task->period;
285  if (now - wake > maxDelayInPeriods*period) {
286  period = period * ((now - wake) / period);
287  }
288  // program next period:
289  // 1. convert period to timespec
290  TIME_SPEC ts = ticks2timespec( nano2ticks(period) );
291  // 2. Add ts to periodMark (danger: tn guards for overflows!)
292  NANO_TIME tn = (task->periodMark.tv_nsec + ts.tv_nsec);
293  task->periodMark.tv_nsec = tn % 1000000000LL;
294  task->periodMark.tv_sec += ts.tv_sec + tn / 1000000000LL;
295  }
296  else
297  {
298  TIME_SPEC ts = ticks2timespec( nano2ticks( task->period) );
299  TIME_SPEC now = ticks2timespec( rtos_get_time_monotonic_ns() );
300  NANO_TIME tn = (now.tv_nsec + ts.tv_nsec);
301  task->periodMark.tv_nsec = tn % 1000000000LL;
302  task->periodMark.tv_sec = ts.tv_sec + now.tv_sec + tn / 1000000000LL;
303  }
304 
305  return now > wake ? -1 : 0;
306  }

315  INTERNAL_QUAL int rtos_task_check_scheduler(int* scheduler)
316  {
317 #ifdef ORO_OS_LINUX_CAP_NG
318  if(capng_get_caps_process()) {
319  log(Error) << "Failed to retrieve capabilities (lowering to SCHED_OTHER)." <<endlog();
320  *scheduler = SCHED_OTHER;
321  return -1;
322  }
323 #endif
324 
325  if (*scheduler != SCHED_OTHER && geteuid() != 0
326 #ifdef ORO_OS_LINUX_CAP_NG
327  && capng_have_capability(CAPNG_EFFECTIVE, CAP_SYS_NICE)==0
328 #endif
329  ) {
330  // they're not root and they want a real-time priority, which _might_
331  // be acceptable if they're using pam_limits and have set the rtprio ulimit
332  // (see "/etc/security/limits.conf" and "ulimit -a")
333  struct rlimit r;
334  if ((0 != getrlimit(RLIMIT_RTPRIO, &r)) || (0 == r.rlim_cur))
335  {
336  log(Warning) << "Lowering scheduler type to SCHED_OTHER for non-privileged users.." <<endlog();
337  *scheduler = SCHED_OTHER;
338  return -1;
339  }
340  }
341 
342  if (*scheduler != SCHED_OTHER && *scheduler != SCHED_FIFO && *scheduler != SCHED_RR ) {
343  log(Error) << "Unknown scheduler type." <<endlog();
344  *scheduler = SCHED_OTHER;
345  return -1;
346  }
347  return 0;
348  }

350  INTERNAL_QUAL int rtos_task_check_priority(int* scheduler, int* priority)
351  {
352  int ret = 0;
353  // check scheduler first.
354  ret = rtos_task_check_scheduler(scheduler);
355 
356  // correct priority
357  if (*scheduler == SCHED_OTHER) {
358  if ( *priority != 0 ) {
359  if (*priority != LowestPriority)
360  log(Warning) << "Forcing priority ("<<*priority<<") of thread with SCHED_OTHER policy to 0." <<endlog();
361  *priority = 0;
362  ret = -1;
363  }
364  } else {
365  // SCHED_FIFO/SCHED_RR:
366  if (*priority <= 0){
367  log(Warning) << "Forcing priority ("<<*priority<<") of thread with !SCHED_OTHER policy to 1." <<endlog();
368  *priority = 1;
369  ret = -1;
370  }
371  if (*priority > 99){
372  log(Warning) << "Forcing priority ("<<*priority<<") of thread with !SCHED_OTHER policy to 99." <<endlog();
373  *priority = 99;
374  ret = -1;
375  }
376  // and limit them according to pam_Limits (only if not root)
377  if ( geteuid() != 0
378 #ifdef ORO_OS_LINUX_CAP_NG
379  && !capng_have_capability(CAPNG_EFFECTIVE, CAP_SYS_NICE)
380 #endif
381  )
382  {
383  struct rlimit r;
384  if (0 == getrlimit(RLIMIT_RTPRIO, &r))
385  {
386  if (*priority > (int)r.rlim_cur)
387  {
388  log(Warning) << "Forcing priority ("<<*priority<<") of thread with !SCHED_OTHER policy to the pam_limit of " << r.rlim_cur <<endlog();
389  *priority = r.rlim_cur;
390  ret = -1;
391  }
392  }
393  else
394  {
395  // this should not be possible, but do something intelligent
396  *priority = 1;
397  ret = -1;
398  }
399  }
400  }
401  return ret;
402  }
