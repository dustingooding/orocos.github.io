192  bool Push( param_t item)
193  {
194  if ( capacity() == (size_type)bufs->size() ) {
195  if (!mcircular) {
196  droppedSamples.inc();
197  return false;
198  }
199  // we will recover below in case of circular
200  }
201  Item* mitem = mpool->allocate();
202  if ( mitem == 0 ) { // queue full ( rare but possible in race with PopWithoutRelease )
203  if (!mcircular) {
204  droppedSamples.inc();
205  return false;
206  }
207  else {
208  if (bufs->dequeue( mitem ) == false ) {
209  droppedSamples.inc();
210  return false; // assert(false) ???
211  }
212  // we keep mitem to write item to next
213  }
214  }
215 
216  // copy over.
217  *mitem = item;
218  if (bufs->enqueue( mitem ) == false ) {
219  //got memory, but buffer is full
220  //this can happen, as the memory pool is
221  //bigger than the buffer
222  if (!mcircular) {
223  mpool->deallocate( mitem );
224  droppedSamples.inc();
225  return false;
226  } else {
227  // pop & deallocate until we have free space.
228  Item* itmp = 0;
229  do {
230  if ( bufs->dequeue( itmp ) ) {
231  mpool->deallocate( itmp );
232  droppedSamples.inc();
233  } else {
234  // Both operations, enqueue() and dequeue() failed on the buffer:
235  // We could free the allocated pool item return false here,
236  // but in fact this can only happen during massive concurrent
237  // access to the circular buffer or in the trivial case that
238  // the buffer size is zero. So just keep on trying...
239  }
240  } while ( bufs->enqueue( mitem ) == false );
241  }
242  }
243  return true;
244  }
