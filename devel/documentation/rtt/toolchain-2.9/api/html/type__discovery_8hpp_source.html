330  template<class T>
331  type_discovery &load_a_type(const boost::serialization::nvp<T> & t, boost::mpl::false_)
332  {
333  // check for single-member extraction first:
334  if ( !membername.empty() ) {
335  // Only serialize if the name matches:
336  if ( t.name() == membername ) {
337  if ( !mref ) {
338  *this & t.value();
339  } else {
340  mref->setReference( (void*) & t.value() );
341  mref = 0; // signals' we're done.
342  }
343  }
344  } else {
345  // Full extraction of all parts:
346  // store name of member
347  mnames.push_back( t.name() );
348 
349  // serialize the data as usual
350  if (mparent)
351  *this & t.value();
352  }
353 
354  return *this;
355  }
