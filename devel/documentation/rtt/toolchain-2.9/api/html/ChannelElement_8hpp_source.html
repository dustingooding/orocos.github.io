164  virtual FlowStatus read(reference_t sample, bool copy_old_data = true)
165  {
166  FlowStatus result = NoData;
167  // read and iterate if necessary.
168 #ifdef USE_CPP11
169  select_reader_channel( bind( &MultipleInputsChannelElement<T>::do_read, this, boost::ref(sample), boost::ref(result), _1, _2), copy_old_data );
170 #else
171  select_reader_channel( boost::bind( &MultipleInputsChannelElement<T>::do_read, this, boost::ref(sample), boost::ref(result), boost::lambda::_1, boost::lambda::_2), copy_old_data );

176  typename ChannelElement<T>::shared_ptr currentInput() {
177  typename ChannelElement<T>::shared_ptr last;
178  BufferPolicy buffer_policy = getBufferPolicy();
179  if (!buffer_policy) buffer_policy = ConnPolicy::Default().buffer_policy;
180  switch(buffer_policy) {
181  case PerConnection:
182  case PerOutputPort:
183  last = last_read;
184  if ( !last && !inputs.empty() ) last = inputs.front()->template narrow<T>();
185  break;
186  case PerInputPort:
187  case Shared:
188  last = last_signalled->narrow<T>();
189  break;
190  default:
191  return typename ChannelElement<T>::shared_ptr();
192  }
193  return last;
194  }
195 
196  private:
197  bool do_read(reference_t sample, FlowStatus& result, bool copy_old_data, typename ChannelElement<T>::shared_ptr& input)
198  {
199  assert( result != NewData );
200  if ( input ) {
201  FlowStatus tresult = input->read(sample, copy_old_data);
202  // the result trickery is for not overwriting OldData with NoData.
203  if (tresult == NewData) {
204  result = tresult;
205  return true;
206  }
207  // stores OldData result
208  if (tresult > result)
209  result = tresult;
210  }
211  return false;
212  }

222  template<typename Pred>
223  void select_reader_channel(Pred pred, bool copy_old_data) {
224  RTT::os::SharedMutexLock lock(inputs_lock);
225  typename ChannelElement<T>::shared_ptr new_input =
226  find_if(pred, copy_old_data);
227 
228  if (new_input)
229  {
230  // We don't clear the current channel (to get it to NoData state), because there is a race
231  // between find_if and this line. We have to accept (in other parts of the code) that eventually,
232  // all channels return 'OldData'.
233  last_read = new_input.get();
234  }
235  }

237  template<typename Pred>
238  typename ChannelElement<T>::shared_ptr find_if(Pred pred, bool copy_old_data) {
239  typename ChannelElement<T>::shared_ptr current = currentInput();
240 
241  // We only copy OldData in the initial read of the current channel.
242  // if it has no new data, the search over the other channels starts,
243  // but no old data is needed.
244  if ( current )
245  if ( pred( copy_old_data, current ) )
246  return current;
247 
248  if (buffer_policy == PerConnection || buffer_policy == PerOutputPort) {
249  for (Inputs::iterator it = inputs.begin(); it != inputs.end(); ++it) {
250  if (*it == current) continue;
251  typename ChannelElement<T>::shared_ptr input = (*it)->narrow<T>();
252  assert(input);
253  if ( pred(false, input) == true)
254  return input;
255  }
256  }
257 
258  return typename ChannelElement<T>::shared_ptr();
259  }

283  virtual WriteStatus data_sample(param_t sample, bool reset = true)
284  {
285  WriteStatus result = WriteSuccess;
286  bool at_least_one_output_is_disconnected = false;
287  bool at_least_one_output_is_connected = false;
288 
289  {
290  RTT::os::SharedMutexLock lock(outputs_lock);
291  {
292  RTT::os::SharedMutexLock lock2(buffer_policy_lock);
293  assert((buffer_policy != PerOutputPort && buffer_policy != Shared) || (outputs.size() <= 1));
294  }
295 
296  if (outputs.empty()) return WriteSuccess;
297  for(Outputs::iterator it = outputs.begin(); it != outputs.end(); ++it)
298  {
299  typename ChannelElement<T>::shared_ptr output = it->channel->narrow<T>();
300  WriteStatus fs = output->data_sample(sample, reset);
301  if (result < fs) result = fs;
302  if (fs == NotConnected) {
303  it->disconnected = true;
304  at_least_one_output_is_disconnected = true;
305  } else {
306  at_least_one_output_is_connected = true;
307  }
308  }
309  }
310 
311  if (at_least_one_output_is_disconnected) {
312  removeDisconnectedOutputs();
313  if (!at_least_one_output_is_connected) result = NotConnected;
314  }
315 
316  return result;
317  }

325  {
326  WriteStatus result = WriteSuccess;
327  bool at_least_one_output_is_disconnected = false;
328  bool at_least_one_output_is_connected = false;
329 
330  {
331  RTT::os::SharedMutexLock lock(outputs_lock);
332  {
333  RTT::os::SharedMutexLock lock2(buffer_policy_lock);
334  assert((buffer_policy != PerOutputPort && buffer_policy != Shared) || (outputs.size() <= 1));
