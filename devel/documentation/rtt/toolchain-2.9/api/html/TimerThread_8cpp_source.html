66  TimerThreadPtr TimerThread::Instance(int scheduler, int pri, double per, unsigned cpu_affinity)
67  {
68  // Since the period is stored as nsecs, we convert per to NS in order
69  // to get a match.
70  os::CheckPriority(scheduler, pri);
71  TimerThreadList::iterator it = TimerThreads.begin();
72  while ( it != TimerThreads.end() ) {
73  TimerThreadPtr tptr = it->lock();
74  // detect old pointer.
75  if ( !tptr ) {
76  TimerThreads.erase(it);
77  it = TimerThreads.begin();
78  continue;
79  }
80  if ( tptr->getScheduler() == scheduler && tptr->getPriority() == pri && tptr->getPeriodNS() == Seconds_to_nsecs(per) ) {
81  return tptr;
82  }
83  ++it;
84  }
85  TimerThreadPtr ret( new TimerThread(scheduler, pri, "TimerThreadInstance", per, cpu_affinity) );
86  TimerThreads.push_back( ret );
87  return ret;
88  }

145  void TimerThread::step() {
146  MutexLock lock(mutex);
147 
148  // The size of the tasks vector does not change during add/remove, thus
149  // t_iter is never invalidated.
150  for( ActivityList::iterator t_iter = tasks.begin(); t_iter != tasks.end(); ++t_iter)
151  if ( *t_iter ) {
152  (*t_iter)->step();
153  if (*t_iter)
154  (*t_iter)->work(RunnableInterface::TimeOut);
155  }
156 
157  if ( cleanup )
158  this->reorderList();
159  }

161  void TimerThread::reorderList() {
162  // reorder the list to remove clear'ed tasks
163  ActivityList::iterator begin = tasks.begin();
164  // first zero :
165  PeriodicActivity* nullActivity = 0;
166  ActivityList::iterator it = tasks.begin();
167 
168  it = find( tasks.begin(), tasks.end(), nullActivity); // First zero
169  begin = it+1;
170  while ( it != tasks.end() ) {
171  // Look for first non-zero after 'it' :
172  while ( begin != tasks.end() && *begin == 0 )
173  ++begin;
174  if ( begin == tasks.end() ) { // if no task found after zero :
175 // Logger::log() << Logger::Error << "beginBefore resize :"<< tasks.size() << Logger::endl;
176  tasks.resize( it - tasks.begin() ); // cut out the items after 'it'
177 // Logger::log() << Logger::Error << "beginAfter resize :"<< tasks.size() << Logger::endl;
178  break; // This is our exit condition !
179  }
180  // first zero after begin :
181  ActivityList::iterator end = find ( begin, tasks.end(), nullActivity);
182  // if end == tasks.end() ==> next while will copy all
183  // if end != tasks.end() ==> next while will copy first zero's
184  while ( begin != end ) {
185  *it = *begin; // copy operation
186  ++begin;
187  ++it; // go to next slot to inspect.
188  }
189  }
190 
191  cleanup = false;
192  }
