96  virtual bool execute() {
97  // this is asyn behaviour :
98  if (isqueued == false ) {
99  isqueued = true;
100  maccept = minit->execute() && mrunner->runFunction( _foo.get() );
101  // we ignore the ret value of start(). It could have been auto-started during loading() of the function.
102  if ( _foo->needsStart() ) // _foo might be auto-started in runFunction()
103  _foo->start();
104  if ( maccept ) {
105  // block for the result: foo stopped or in error
106  //mcaller->waitForFunctions(boost::bind(&CallFunction::fooDone,this) );
107  mrunner->waitForFunctions(boost::bind(&CallFunction::fooDone,this) );
108  if ( _foo->inError() ) {
109  throw false;
110  }
111  return true;
112  }
113  return false;
114  }
115  return true;
116  }
