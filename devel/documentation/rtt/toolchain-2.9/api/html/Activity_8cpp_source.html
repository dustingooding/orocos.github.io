172  void Activity::loop() {
173  nsecs wakeup = 0;
174  int overruns = 0;
175  while ( true ) {
176  // since update_period may be changed at any time, we need to recheck it each time:
177  if ( update_period > 0.0) {
178  // initialize first wakeup time if we are periodic.
179  if ( wakeup == 0 ) {
180  wakeup = os::TimeService::Instance()->getNSecs() + Seconds_to_nsecs(update_period);
181  }
182  } else {
183  // only clear if update_period <= 0.0 :
184  wakeup = 0;
185  }
186 
187  // periodic: we flag mtimeout below; non-periodic: we flag mtimeout in timeout()
188  if (mtimeout) {
189  // was a timeout() call, or internally generated after wakeup
190  mtimeout = false;
191  this->step();
192  this->work(base::RunnableInterface::TimeOut);
193  } else {
194  // was a trigger() call
195  if ( update_period > 0 ) {
196  this->step();
197  this->work(base::RunnableInterface::Trigger);
198  } else {
199  if (runner) {
200  runner->loop();
201  runner->work(base::RunnableInterface::Trigger);
202  } else {
203  this->step();
204  this->work(base::RunnableInterface::Trigger);
205  }
206  }
207  // if a timeout() was done during work(), we will re-enter
208  // loop() due to the Thread::start().
209  }
210  // next, sleep/wait
211  os::MutexLock lock(msg_lock);
212  if ( wakeup == 0 ) {
213  // non periodic, default behavior:
214  return;
215  } else {
216  // If periodic, sleep until wakeup time or a message comes in.
217  // when wakeup time passed, wait_until will return false and we recalculate wakeup + mtimeout
218  bool time_elapsed = ! msg_cond.wait_until(msg_lock,wakeup);
219 
220  if (time_elapsed) {
221  nsecs nsperiod = Seconds_to_nsecs(update_period);
222  wakeup = wakeup + nsperiod;
223  // calculate next wakeup point, overruns causes skips:
224  nsecs now = os::TimeService::Instance()->getNSecs();
225  if ( wakeup < now )
226  {
227  ++overruns;
228  if (overruns == maxOverRun)
229  break; // break while(true)
230  }
231  else if (overruns != 0) {
232  --overruns;
233  }
234  if ( !mabswaitpolicy && wakeup < now ) {
235  wakeup = wakeup + ((now-wakeup)/nsperiod+1)*nsperiod; // assumes that (now-wakeup)/nsperiod rounds down !
236  }
237  mtimeout = true;
238  }
239  }
240  if (mstopRequested) {
241  mstopRequested = false; // guarded by Mutex lock
242  return;
243  }
244  }
245  if (overruns == maxOverRun)
246  {
247  this->emergencyStop();
248  log(Critical) << rtos_task_get_name(this->getTask())
249  << " got too many periodic overruns in step() ("
250  << overruns << " times), stopped Thread !"
251  << endlog();
252  log(Critical) << " See Thread::setMaxOverrun() for info."
253  << endlog();
254  }
255  }

274  bool Activity::stop()
275  {
276  if (!active)
277  return false;
278 
279  running = false;
280 
281  // first of all, exit loop() if possible:
282  {
283  os::MutexLock lock(msg_lock); // protects stopRequested
284  mstopRequested = true;
285  msg_cond.broadcast();
286  }
287 
288  if (update_period == 0)
289  {
290  if ( inloop ) {
291  if ( !this->breakLoop() ) {
292  log(Warning) << "Failed to stop thread " << this->getName() << ": breakLoop() returned false."<<endlog();
293  running = true;
294  return false;
295  }
296  // breakLoop was ok, wait for loop() to return.
297  }
298  MutexTimedLock lock(breaker, getStopTimeout());
299  if ( !lock.isSuccessful() ) {
300  log(Error) << "Failed to stop thread " << this->getName() << ": breakLoop() returned true, but loop() function did not return after "<<getStopTimeout() << " second(s)."<<endlog();
301  running = true;
302  return false;
303  }
304  } else {
305  //
306  MutexTimedLock lock(breaker, getStopTimeout() );
307  if ( lock.isSuccessful() ) {
308  // drop out of periodic mode.
309  rtos_task_make_periodic(&rtos_task, 0);
310  } else {
311  log(Error) << "Failed to stop thread " << this->getName() << ": step() function did not return after "<< getStopTimeout() <<" second(s)."<<endlog();
312  running = true;
313  return false;
314  }
315  }
316 
317  this->finalize();
318  active = false;
319  return true;
320  }
