104  void ExecutionEngine::processFunctions()
105  {
106  // Execute all loaded Functions :
107  ExecutableInterface* foo = 0;
108  int nbr = f_queue->size(); // nbr to process.
109  // 1. Fetch new ones from queue.
110  while ( f_queue->dequeue(foo) ) {
111  assert(foo);
112  if ( foo->execute() == false ){
113  foo->unloaded();
114  os::MutexLock lock(msg_lock);
115  msg_cond.broadcast(); // required for waitForFunctions() (3rd party thread)
116  } else {
117  f_queue->enqueue( foo );
118  }
119  if ( --nbr == 0) // we did a round-trip
120  break;
121  }
122  }

152  bool ExecutionEngine::removeFunction( ExecutableInterface* f )
153  {
154  // Remove from the queue.
155  if ( !f )
156  return false;
157 
158  if ( !f->isLoaded() )
159  return true;
160 
161  // When not running, just remove.
162  if ( !this->getActivity()->isActive() ) {
163  if ( removeSelfFunction( f ) == false )
164  return false;
165  } else {
166  // Running: create message on stack.
167  RemoveMsg rmsg(f,this);
168  if ( this->process(&rmsg) )
169 #ifdef USE_CPP11
170  this->waitForMessages( ! bind(&ExecutableInterface::isLoaded, f) || bind(&RemoveMsg::found,boost::ref(rmsg)) );
171 #else
172  this->waitForMessages( ! lambda::bind(&ExecutableInterface::isLoaded, f) || lambda::bind(&RemoveMsg::found,boost::ref(rmsg)) );
173 #endif
174  if (!rmsg.found)
175  return false;
176  }
177  // unloading was succesful, now notify unloading:
178  f->unloaded();
179  return true;
180  }

182  bool ExecutionEngine::removeSelfFunction(ExecutableInterface* f )
183  {
184  // since this function is executed in process messages, it is always safe to execute.
185  if ( !f )
186  return false;
187  int nbr = f_queue->size();
188  while (nbr != 0) {
189  ExecutableInterface* foo = 0;
190  if ( !f_queue->dequeue(foo) )
191  return false;
192  if ( f == foo) {
193  return true;
194  }
195  f_queue->enqueue(foo);
196  --nbr;
197  }
198  return true;
199  }

210  void ExecutionEngine::processMessages()
211  {
212  // Fast bail-out :
213  if ( mqueue->isEmpty() )
214  return;
215  // execute all commands from the AtomicQueue.
216  // msg_lock may not be held when entering this function !
217  DisposableInterface* com(0);
218  {
219  while ( mqueue->dequeue(com) ) {
220  assert( com );
221  com->executeAndDispose();
222  }
223  }
224 
225  // there's no need to hold the lock during
226  // emptying the queue. But we must hold the
227  // lock once between excuteAndDispose and the
228  // broadcast to avoid the race condition in
229  // waitForMessages().
230  // This allows us to recurse into processMessages.
231  MutexLock locker( msg_lock );
232  msg_cond.broadcast(); // required for waitForMessages() (3rd party thread)
233  }

253  bool ExecutionEngine::process( DisposableInterface* c )
254  {
255  // We only reject running functions when we're in the FatalError state.
256  if (taskc && taskc->mTaskState == TaskCore::FatalError )
257  return false;
258 
259  // forward message to master ExecutionEngine if available
260  if (mmaster) {
261  return mmaster->process(c);
262  }
263 
264  if ( c && this->getActivity() ) {
265  bool result = mqueue->enqueue( c );
266  this->getActivity()->trigger();
267  {
268  os::MutexLock lock(msg_lock);
269  msg_cond.broadcast(); // required for waitAndProcessMessages() (EE thread)
270  }
271  return result;
272  }
273  return false;
274  }

335  void ExecutionEngine::waitForMessagesInternal(boost::function<bool(void)> const& pred)
336  {
337  // Note: waitForMessagesInternal() can be called from waitForFunctions even if this is a slave engine!
338  // assert( mmaster == 0 );
339  if ( pred() )
340  return;
341  // only to be called from the thread not executing step().
342  os::MutexLock lock(msg_lock);
343  while (!pred()) { // the mutex guards that processMessages can not run between !pred and the wait().
344  msg_cond.wait(msg_lock); // now processMessages may run.
345  }

349  void ExecutionEngine::waitAndProcessMessages(boost::function<bool(void)> const& pred)
350  {
351  assert( mmaster == 0 );
352  while ( !pred() ){
353  // may not be called while holding the msg_lock !!!
354  this->processMessages();
355  {
356  // only to be called from the thread executing step().
357  // We must lock because the cond variable will unlock msg_lock.
358  os::MutexLock lock(msg_lock);
359  if (!pred()) {
360  msg_cond.wait(msg_lock); // now processMessages may run.
361  } else {
362  return; // do not process messages when pred() == true;
363  }
364  }
365  }
366  }

390  void ExecutionEngine::work(RunnableInterface::WorkReason reason) {
391  // Interprete work before calling into user code such that we are consistent at all times.
392  if (taskc) {
393  ++taskc->mCycleCounter;
394  switch(reason) {
395  case RunnableInterface::Trigger :
396  ++taskc->mTriggerCounter;
397  break;
398  case RunnableInterface::TimeOut :
399  ++taskc->mTimeOutCounter;
400  break;
401  case RunnableInterface::IOReady :
402  ++taskc->mIOCounter;
403  break;
404  default:
405  break;
406  }
407  }
408  if (reason == RunnableInterface::Trigger) {
409  /* Callback step */
410  processMessages();
411  processPortCallbacks();
412  } else if (reason == RunnableInterface::TimeOut || reason == RunnableInterface::IOReady) {
413  /* Update step */
414  processMessages();
415  processPortCallbacks();
416  processFunctions();
417  processHooks();
418  }
419  }
420  void ExecutionEngine::processHooks() {
421  // only call updateHook in the Running state.
422  if ( taskc ) {
423  // A trigger() in startHook() will be ignored, we trigger in TaskCore after startHook finishes.
424  if ( taskc->mTaskState == TaskCore::Running && taskc->mTargetState == TaskCore::Running ) {
425  TRY (
426  taskc->updateHook();
427  ) CATCH(std::exception const& e,
428  log(Error) << "in updateHook(): switching to exception state because of unhandled exception" << endlog();
429  log(Error) << " " << e.what() << endlog();
430  taskc->exception();
431  ) CATCH_ALL (
432  log(Error) << "in updateHook(): switching to exception state because of unhandled exception" << endlog();
433  taskc->exception(); // calls stopHook,cleanupHook
434  )
435  }
436  // in case start() or updateHook() called error(), this will be called:
437  if ( taskc->mTaskState == TaskCore::RunTimeError ) {
438  TRY (
439  taskc->errorHook();
440  ) CATCH(std::exception const& e,
441  log(Error) << "in errorHook(): switching to exception state because of unhandled exception" << endlog();
442  log(Error) << " " << e.what() << endlog();
443  taskc->exception();
444  ) CATCH_ALL (
445  log(Error) << "in errorHook(): switching to exception state because of unhandled exception" << endlog();
446  taskc->exception(); // calls stopHook,cleanupHook
447  )
448  }
449  }
