106  bool Push( param_t item )
107  {
108  if (cap == (size_type)buf.size() ) {
109  if (!mcircular)
110  return false;
111  else
112  buf.pop_front();
113  }
114  buf.push_back( item );
115  return true;
116  }

118  size_type Push(const std::vector<T>& items)
119  {
120  typename std::vector<T>::const_iterator itl( items.begin() );
121  if (mcircular && (size_type)items.size() >= cap ) {
122  // clear out current data and reset iterator to first element we're going to take.
123  buf.clear();
124  itl = items.begin() + ( items.size() - cap );
125  } else if ( mcircular && (size_type)(buf.size() + items.size()) > cap) {
126  // drop excess elements from front
127  assert( (size_type)items.size() < cap );
128  while ( (size_type)(buf.size() + items.size()) > cap )
129  buf.pop_front();
130  // itl still points at first element of items.
131  }
132  while ( ((size_type)buf.size() != cap) && (itl != items.end()) ) {
133  buf.push_back( *itl );
134  ++itl;
135  }
136  return (itl - items.begin());
137  }
