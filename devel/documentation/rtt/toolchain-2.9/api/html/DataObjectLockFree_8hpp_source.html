189  virtual FlowStatus Get( DataType& pull, bool copy_old_data = true ) const
190  {
191  if (!initialized) {
192  return NoData;
193  }
194 
195  PtrType reading;
196  // loop to combine Read/Modify of counter
197  // This avoids a race condition where read_ptr
198  // could become write_ptr ( then we would read corrupted data).
199  do {
200  reading = read_ptr; // copy buffer location
201  oro_atomic_inc(&reading->counter); // lock buffer, no more writes
202  // XXX smp_mb
203  if ( reading != read_ptr ) // if read_ptr changed,
204  oro_atomic_dec(&reading->counter); // better to start over.
205  else
206  break;
207  } while ( true );
208  // from here on we are sure that 'reading'
209  // is a valid buffer to read from.
210 
211  FlowStatus result = reading->status;
212  if (result == NewData) {
213  pull = reading->data; // takes some time
214  reading->status = OldData; // CAS?
215  } else if ((result == OldData) && copy_old_data) {
216  pull = reading->data; // takes some time
217  }
218 
219  // XXX smp_mb
220  oro_atomic_dec(&reading->counter); // release buffer
221  return result;
222  }

229  virtual bool Set( const DataType& push )
230  {
240  if (!initialized) {
241  types::TypeInfo *ti = types::Types()->getTypeInfo<DataType>();
242  log(Error) << "You set a lock-free data object of type " << (ti ? ti->getTypeName() : "(unknown)") << " without initializing it with a data sample. "
243  << "This might not be real-time safe." << endlog();
244  data_sample(DataType(), true);
245  }
246 
247  // writeout in any case
248  PtrType writing = write_ptr;
249  writing->data = push;
250  writing->status = NewData;
251 
252  // if next field is occupied (by read_ptr or counter),
253  // go to next and check again...
254  while ( oro_atomic_read( &write_ptr->next->counter ) != 0 || write_ptr->next == read_ptr )
255  {
256  write_ptr = write_ptr->next;
257  if (write_ptr == writing)
258  return false; // nothing found, to many readers !
259  }
260 
261  // we will be able to move, so replace read_ptr
262  read_ptr = writing;
263  write_ptr = write_ptr->next; // we checked this in the while loop
264  return true;
265  }

268  if (!initialized || reset) {
269  // prepare the buffer.

271  data[i].data = sample;

275  data[BUF_LEN-1].next = &data[0];
276  initialized = true;
277  return true;
278  } else {
279  return initialized;
280  }
