73 bool typeDecomposition( base::DataSourceBase::shared_ptr dsb, PropertyBag& targetbag, bool recurse)
74 {
75  if (!dsb)
76  return false;
77 
78  // try user's custom type decomposition first:
79  DataSourceBase::shared_ptr decomposed = dsb->getTypeInfo()->decomposeType(dsb);
80  // In cases where decomposeType() returned dsb itself, we stop the decomposition here.
81  if (decomposed == dsb)
82  return false;
83  if (decomposed) {
84  // decomposed is or another type, or a PropertyBag
85  internal::AssignableDataSource<PropertyBag>::shared_ptr bag = internal::AssignableDataSource<PropertyBag>::narrow( decomposed.get() );
86  if ( bag ) {
87  // get it and copy it.
88  targetbag = bag->rvalue();
89  return true;
90  } else {
91  // it converted to something else than a bag.
92  //log(Debug) << "propertyDecomposition: decomposeType() of "<< dsb->getTypeName() << " did not return a PropertyBag but a " << decomposed->getTypeName() << endlog();
93  return false;
94  }
95  }
96 
97  vector<string> parts = dsb->getMemberNames();
98  if ( parts.empty() ) {
99  log(Debug) << "propertyDecomposition: "<< dsb->getTypeName() << " does not have any members." << endlog();
100  return false;
101  }
102 
103  targetbag.setType( dsb->getTypeName() );
104 
105  // needed for recursion.
106  auto_ptr< Property<PropertyBag> > recurse_bag( new Property<PropertyBag>("recurse_bag","Part") );
107  // First at the explicitly listed parts:
108  for(vector<string>::iterator it = parts.begin(); it != parts.end(); ++it ) {
109  DataSourceBase::shared_ptr part = dsb->getMember( *it );
110  if (!part) {
111  log(Error) <<"propertyDecomposition: Inconsistent type info for "<< dsb->getTypeName() << ": reported to have part '"<<*it<<"' but failed to return it."<<endlog();
112  continue;
113  }
114  if ( !part->isAssignable() ) {
115  // For example: the case for size() and capacity() in SequenceTypeInfo
116  log(Debug)<<"propertyDecomposition: Part "<< *it << ":"<< part->getTypeName() << " is not changeable."<<endlog();
117  continue;
118  }
119  // finally recurse or add it to the target bag:
120  PropertyBase* newpb = part->getTypeInfo()->buildProperty(*it,"Part",part);
121  if ( !newpb ) {
122  log(Error)<< "Decomposition failed because Part '"<<*it<<"' is not known to type system."<<endlog();
123  continue;
124  }
125  if ( !recurse )
126  targetbag.ownProperty( newpb ); // leaf
127  else if ( !propertyDecomposition( newpb, recurse_bag->value(), true) ) {
128  //recurse_bag is empty
129  assert( recurse_bag->rvalue().empty() );
130  base::DataSourceBase::shared_ptr clone = newpb->getDataSource();
131  base::DataSourceBase::shared_ptr converted = clone->getTypeInfo()->decomposeType(clone);
132  if ( converted && converted != clone ) {
133  // converted contains another type
134  targetbag.add( converted->getTypeInfo()->buildProperty(*it, "", converted) );
135  } else {
136  // use the original
137  targetbag.add( newpb );
138  }
139  } else {
140  assert( recurse_bag->rvalue().size() >= 1 );
141  recurse_bag->setName(*it);
142  // setType() is done by recursive of self.
143  targetbag.ownProperty( recurse_bag.release() ); //recursed.
144  recurse_bag.reset( new Property<PropertyBag>("recurse_bag","Part") );
145  delete newpb; // since we recursed, the recurse_bag now 'embodies' newpb.
146  }
147  }
148 
149  // Prepare recurse_bag for storing composite items:
150  recurse_bag->setDescription("Item");
151 
152  // Next get the numbered parts:
153  DataSource<int>::shared_ptr size = DataSource<int>::narrow( dsb->getMember("size").get() );
154  if (size) {
155  int msize = size->get();
156  for (int i=0; i < msize; ++i) {
157  string indx = boost::lexical_cast<string>( i );
158  DataSourceBase::shared_ptr item = dsb->getMember(indx);
159  if (item) {
160  if ( !item->isAssignable() ) {
161  // For example: the case for size() and capacity() in SequenceTypeInfo
162  log(Warning)<<"propertyDecomposition: Item '"<< indx << "' of type "<< dsb->getTypeName() << " is not changeable."<<endlog();
163  continue;
164  }
165  // finally recurse or add it to the target bag:
166  PropertyBase* newpb = item->getTypeInfo()->buildProperty( "Element" + indx,"Sequence Element",item);
167  if ( !recurse || !propertyDecomposition( newpb, recurse_bag->value()) ) {
168  targetbag.ownProperty( newpb ); // leaf
169  } else {
170  delete newpb;
171  recurse_bag->setName( "Element" + indx );
172  // setType() is done by recursive of self.
173  targetbag.ownProperty( recurse_bag.release() ); //recursed.
174  recurse_bag.reset( new Property<PropertyBag>("recurse_bag","Item") );
175  }
176  }
177  }
178  }
179  if (targetbag.empty() )
180  log(Debug) << "propertyDecomposition: "<< dsb->getTypeName() << " returns an empty property bag." << endlog();
181  return true;
182 }
